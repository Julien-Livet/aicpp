def dsl1(I):
    # General horizontal double-breaker: for any color c, if there is a horizontal "cc",
    # recolor the right/left member to match its adjacent non-c neighbor.
    G = I
    pal = totuple(palette(G))
    for c in pal:
        oc = ofcolor(G, c)
        # right cell of a horizontal double "cc"
        right_double = intersection(oc, shift(oc, LEFT))
        # left cell of a horizontal double "cc"
        left_double = intersection(oc, shift(oc, RIGHT))
        for d in pal:
            if d == c:
                continue
            od = ofcolor(G, d)
            # if right cell of "cc" has a right neighbor of color d -> set it to d
            t1 = intersection(right_double, shift(od, LEFT))
            # if left cell of "cc" has a left neighbor of color d -> set it to d
            t2 = intersection(left_double, shift(od, RIGHT))
            G = fill(G, d, t1)
            G = fill(G, d, t2)
    return G

def dsl2(I):
    # Parity propagation by two steps within non-separator zones (avoid 2 and 4 as separators).
    G = I
    pal = totuple(palette(G))
    sep = combine(ofcolor(G, TWO), ofcolor(G, FOUR))
    allowed = difference(asindices(G), sep)
    for d in pal:
        if d == TWO or d == FOUR:
            continue
        od = ofcolor(G, d)
        # propagate d two to the right where the in-between is allowed
        tR = intersection(shift(od, (0, 2)), allowed)
        tR = intersection(tR, shift(allowed, RIGHT))  # ensure middle is allowed
        # propagate d two to the left where the in-between is allowed
        tL = intersection(shift(od, (0, -2)), allowed)
        tL = intersection(tL, shift(allowed, LEFT))   # ensure middle is allowed
        G = fill(G, d, tR)
        G = fill(G, d, tL)
    return G

def dsl3(I):
    # Conditional parity copy: if left-2 is d but left-1 is not d, set current to d.
    # Symmetric for right-2. Avoid separators 2 and 4 as targets.
    G = I
    pal = totuple(palette(G))
    sep = combine(ofcolor(G, TWO), ofcolor(G, FOUR))
    allowed = difference(asindices(G), sep)
    for d in pal:
        od = ofcolor(G, d)
        # positions two to the right of d
        posR = shift(od, (0, 2))
        notd_left = difference(asindices(G), shift(od, RIGHT))  # left neighbor != d
        tR = intersection(intersection(posR, notd_left), allowed)
        # positions two to the left of d
        posL = shift(od, (0, -2))
        notd_right = difference(asindices(G), shift(od, LEFT))  # right neighbor != d
        tL = intersection(intersection(posL, notd_right), allowed)
        G = fill(G, d, tR)
        G = fill(G, d, tL)
    return G

def dsl4(I):
    # Focused fix for 4-doubles: replace "44" by copying adjacent non-4 inward on both sides.
    G = I
    pal = totuple(palette(G))
    idx4 = ofcolor(G, FOUR)
    # right cell of "44"
    right44 = intersection(idx4, shift(idx4, LEFT))
    # left cell of "44"
    left44 = intersection(idx4, shift(idx4, RIGHT))
    for d in pal:
        if d == FOUR:
            continue
        od = ofcolor(G, d)
        # Break "44" by pulling in the adjacent non-4 colors
        t1 = intersection(right44, shift(od, LEFT))   # ... 4,4,d -> set middle-right 4 to d
        t2 = intersection(left44, shift(od, RIGHT))   # d,4,4 ... -> set middle-left 4 to d
        G = fill(G, d, t1)
        G = fill(G, d, t2)
    return G
