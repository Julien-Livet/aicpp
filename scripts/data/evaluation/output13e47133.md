def dsl1(I):
    w = width(I)
    tl = index(I, astuple(ZERO, ZERO))
    tr = index(I, astuple(ZERO, subtract(w, ONE)))
    L = lefthalf(I)
    R = righthalf(I)
    L2 = replace(L, mostcolor(L), tl)
    R2 = replace(R, mostcolor(R), tr)
    O = hconcat(L2, R2)
    return O

def dsl2(I):
    h = height(I)
    w = width(I)
    tr = index(I, astuple(ZERO, subtract(w, ONE)))
    bl = index(I, astuple(subtract(h, ONE), ZERO))
    T = tophalf(I)
    B = bottomhalf(I)
    T2 = replace(T, mostcolor(T), tr)
    B2 = replace(B, mostcolor(B), bl)
    O = vconcat(T2, B2)
    return O

def dsl4(I):
    w = width(I)
    a = index(I, astuple(ZERO, ZERO))
    b = index(I, astuple(ZERO, subtract(w, ONE)))
    O = switch(I, a, b)
    return O
