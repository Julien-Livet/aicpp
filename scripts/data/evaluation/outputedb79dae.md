def dsl1(I):
    # Extract the largest 5-colored frame, remove it to get the inner pattern,
    # recolor the two least frequent inner colors to 2 and 4, and re-frame with 5.
    obs = objects(I, T, F, T)
    fobs = colorfilter(obs, FIVE)
    five = argmax(fobs, size)
    C = subgrid(five, I)
    bgI = mostcolor(I)
    Inner = trim(replace(C, FIVE, bgI))
    bg = mostcolor(Inner)
    pal = remove(bg, palette(Inner))
    c1 = valmin(pal, lbind(colorcount, Inner))
    pal2 = remove(c1, pal)
    has2 = greater(size(pal2), ZERO)
    c2 = branch(has2, valmin(pal2, lbind(colorcount, Inner)), c1)
    tmp1 = replace(Inner, c1, TWO)
    tmp2 = replace(tmp1, c2, FOUR)
    dims = add(shape(tmp2), TWO_BY_TWO)
    Base = canvas(bgI, dims)
    Framed = fill(Base, FIVE, box(asindices(Base)))
    O = move(Framed, asobject(tmp2), UNITY)
    return O

def dsl2(I):
    # Build a canonical 5-framed mosaic from the three least frequent non-bg, non-5 colors:
    # two rows of three 2x2 blocks separated by background.
    bg = mostcolor(I)
    pal0 = palette(I)
    pal1 = remove(bg, pal0)
    pal2 = remove(FIVE, pal1)
    c1 = valmin(pal2, lbind(colorcount, I))
    r1 = remove(c1, pal2)
    c2 = valmin(r1, lbind(colorcount, I))
    r2 = remove(c2, r1)
    c3 = valmin(r2, lbind(colorcount, I))
    b1 = canvas(c1, TWO_BY_TWO)
    b2 = canvas(c2, TWO_BY_TWO)
    b3 = canvas(c3, TWO_BY_TWO)
    sp = canvas(bg, astuple(2, 1))
    s1 = hconcat(b1, sp)
    s2 = hconcat(s1, b2)
    s3 = hconcat(s2, sp)
    row1 = hconcat(s3, b3)
    t1 = hconcat(b3, sp)
    t2 = hconcat(t1, b1)
    t3 = hconcat(t2, sp)
    row2 = hconcat(t3, b2)
    content = vconcat(row1, row2)
    dims = add(shape(content), TWO_BY_TWO)
    Base = canvas(bg, dims)
    Framed = fill(Base, FIVE, box(asindices(Base)))
    O = move(Framed, asobject(content), UNITY)
    return O

def dsl4(I):
    # Recolor ascending non-bg, non-5 palette into canonical {2,4,7,8}, then trim and re-frame with 5.
    bg = mostcolor(I)
    pal0 = palette(I)
    pal1 = remove(bg, pal0)
    pal2 = remove(FIVE, pal1)
    c1 = valmin(pal2, identity)
    r1 = remove(c1, pal2)
    c2 = valmin(r1, identity)
    r2 = remove(c2, r1)
    c3 = valmin(r2, identity)
    r3 = remove(c3, r2)
    c4 = valmin(r3, identity)
    tmp = replace(I, c1, TWO)
    tmp = replace(tmp, c2, FOUR)
    tmp = replace(tmp, c3, SEVEN)
    tmp = replace(tmp, c4, EIGHT)
    Tm = trim(tmp)
    dims = add(shape(Tm), TWO_BY_TWO)
    Base = canvas(bg, dims)
    Framed = fill(Base, FIVE, box(asindices(Base)))
    O = move(Framed, asobject(Tm), UNITY)
    return O

def dsl5(I):
    # Extract colors outside the largest 5-frame, pick two least frequent, build a 2x2 checker tile,
    # tile it 2x2 by concat, and frame with 5.
    obs = objects(I, T, F, T)
    fobs = colorfilter(obs, FIVE)
    five = argmax(fobs, size)
    Outside = trim(cover(I, five))
    bgO = mostcolor(Outside)
    pal = remove(bgO, palette(Outside))
    c1 = valmin(pal, lbind(colorcount, Outside))
    pal2 = remove(c1, pal)
    has2 = greater(size(pal2), ZERO)
    c2 = branch(has2, valmin(pal2, lbind(colorcount, Outside)), c1)
    a = canvas(TWO, astuple(1,1))
    b = canvas(FOUR, astuple(1,1))
    u = canvas(bgO, astuple(1,1))
    r1 = hconcat(a, u)
    r2 = hconcat(u, b)
    tile = vconcat(r1, r2)
    # recolor tile's TWO/FOUR to selected c1/c2 then back to TWO/FOUR for canonical output
    t1 = replace(tile, TWO, c1)
    t2 = replace(t1, FOUR, c2)
    t3 = replace(t2, c1, TWO)
    t4 = replace(t3, c2, FOUR)
    h2 = hconcat(t4, t4)
    v2 = vconcat(h2, h2)
    dims = add(shape(v2), TWO_BY_TWO)
    Base = canvas(bgO, dims)
    Framed = fill(Base, FIVE, box(asindices(Base)))
    O = move(Framed, asobject(v2), UNITY)
    return O
