def dsl1(I):
    # Unify the two most common background colors, then crop along the dimension that reduces more.
    colors = palette(I)
    countf = lbind(colorcount, I)
    c1 = argmax(colors, countf)
    colors2 = remove(c1, colors)
    c2 = argmax(colors2, countf)
    x = replace(I, c2, c1)
    S = asindices(x)
    BG = ofcolor(x, c1)
    FG = difference(S, BG)
    top = uppermost(FG)
    bot = lowermost(FG)
    L = leftmost(FG)
    R = rightmost(FG)
    H = height(I)
    W = width(I)
    hspan = add(ONE, subtract(bot, top))
    wspan = add(ONE, subtract(R, L))
    dh = subtract(H, hspan)
    dw = subtract(W, wspan)
    croprows = crop(x, astuple(top, 0), astuple(hspan, W))
    cropcols = crop(x, astuple(0, L), astuple(H, wspan))
    O = branch(greater(dh, dw), croprows, cropcols)
    return O

def dsl2(I):
    # Unify backgrounds; choose to crop rows if full width is needed but height can shrink, else crop columns.
    colors = palette(I)
    countf = lbind(colorcount, I)
    c1 = argmax(colors, countf)
    colors2 = remove(c1, colors)
    c2 = argmax(colors2, countf)
    x = replace(I, c2, c1)
    S = asindices(x)
    FG = difference(S, ofcolor(x, c1))
    top = uppermost(FG)
    bot = lowermost(FG)
    L = leftmost(FG)
    R = rightmost(FG)
    H = height(I)
    W = width(I)
    hspan = add(ONE, subtract(bot, top))
    wspan = add(ONE, subtract(R, L))
    fullh = equality(hspan, H)
    fullw = equality(wspan, W)
    croprows = crop(x, astuple(top, 0), astuple(hspan, W))
    cropcols = crop(x, astuple(0, L), astuple(H, wspan))
    # Priority: if only height can shrink keep width -> croprows; if only width can shrink keep height -> cropcols; else pick larger reduction
    dh = subtract(H, hspan)
    dw = subtract(W, wspan)
    prefer_rows = both(flip(fullw), fullh)  # only height can shrink
    prefer_cols = both(flip(fullh), fullw)  # only width can shrink
    tmp = branch(prefer_rows, croprows, branch(prefer_cols, cropcols, branch(greater(dh, dw), croprows, cropcols)))
    return tmp

def dsl5(I):

    fullh = equality(hspan, H)
