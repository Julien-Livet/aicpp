def dsl1(I):
    # Keep large rectangular frames and add a solid center horizontal line inside each
    bg = mostcolor(I)
    O = canvas(bg, shape(I))
    for obj in objects(I, True, False, True):
        if size(obj) > 30:
            col = color(obj)
            O = paint(O, obj)
            inner = inbox(obj)
            if size(inner) == 0:
                continue
            c = center(inner)
            hr = hfrontier(c)
            O = fill(O, col, intersection(hr, inner))
    return O

def dsl3(I):
    # Keep large rectangular frames and add an alternating (every 2 columns) pattern on the center horizontal line
    bg = mostcolor(I)
    O = canvas(bg, shape(I))
    H, W = shape(I)
    for obj in objects(I, True, False, True):
        if size(obj) > 30:
            col = color(obj)
            O = paint(O, obj)
            inner = inbox(obj)
            if size(inner) == 0:
                continue
            c = center(inner)
            hr = hfrontier(c)
            v0 = vfrontier(c)
            # build vertical stripe family every 2 columns, intersect with the center row to get alternating points
            for k in range(-W, W+1, 2):
                vk = shift(v0, astuple(0, k))
                O = fill(O, col, intersection(intersection(vk, hr), inner))
    return O

def dsl5(I):
    # Keep large frames and add a sparse dot grid (every 2 rows and 2 columns) inside their interior
    bg = mostcolor(I)
    O = canvas(bg, shape(I))
    H, W = shape(I)
    for obj in objects(I, True, False, True):
        if size(obj) > 30:
            col = color(obj)
            O = paint(O, obj)
            inner = inbox(obj)
            if size(inner) == 0:
                continue
            c = center(inner)
            hr0 = hfrontier(c)
            vr0 = vfrontier(c)
            for di in range(-H, H+1, 2):
                hr = shift(hr0, astuple(di, 0))
                for dj in range(-W, W+1, 2):
                    vr = shift(vr0, astuple(0, dj))
                    dot = intersection(hr, vr)
                    O = fill(O, col, intersection(dot, inner))
    return O
