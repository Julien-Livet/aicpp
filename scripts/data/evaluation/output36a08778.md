def dsl1(I):
    # Fill entire columns just outside the left/right edges of BLUE runs,
    # and entire rows just outside the top/bottom edges of BLUE columns.
    H, W = shape(I)
    cols_to_fill = set()
    rows_to_fill = set()
    # Per-row BLUE extents -> neighbor columns
    for r in range(H):
        cols = [c for c in range(W) if index(I, (r, c)) == TWO]
        if cols:
            L, R = min(cols), max(cols)
            if L - 1 >= 0:
                cols_to_fill.add(L - 1)
            if R + 1 < W:
                cols_to_fill.add(R + 1)
    # Per-column BLUE extents -> neighbor rows
    for c in range(W):
        rows = [r for r in range(H) if index(I, (r, c)) == TWO]
        if rows:
            U, D = min(rows), max(rows)
            if U - 1 >= 0:
                rows_to_fill.add(U - 1)
            if D + 1 < H:
                rows_to_fill.add(D + 1)
    O = I
    patch = set()
    for r in rows_to_fill:
        for c in range(W):
            if index(O, (r, c)) == SEVEN:
                patch.add((r, c))
    for c in cols_to_fill:
        for r in range(H):
            if index(O, (r, c)) == SEVEN:
                patch.add((r, c))
    if patch:
        O = fill(O, SIX, frozenset(patch))
    return O

def dsl3(I):
    # Combine: columns/rows adjacent to BLUE plus rows/cols containing ORANGE seeds.
    H, W = shape(I)
    rows = set()
    cols = set()
    # Adjacent to BLUE (one step away)
    for r in range(H):
        for c in range(W):
            if index(I, (r, c)) == TWO:
                if r - 1 >= 0 and index(I, (r - 1, c)) == SEVEN:
                    rows.add(r - 1)
                if r + 1 < H and index(I, (r + 1, c)) == SEVEN:
                    rows.add(r + 1)
                if c - 1 >= 0 and index(I, (r, c - 1)) == SEVEN:
                    cols.add(c - 1)
                if c + 1 < W and index(I, (r, c + 1)) == SEVEN:
                    cols.add(c + 1)
    # Include rows/cols that already contain ORANGE
    for r in range(H):
        for c in range(W):
            if index(I, (r, c)) == SIX:
                rows.add(r)
                cols.add(c)
    O = I
    patch = set()
    for r in rows:
        for c in range(W):
            if index(O, (r, c)) == SEVEN:
                patch.add((r, c))
    for c in cols:
        for r in range(H):
            if index(O, (r, c)) == SEVEN:
                patch.add((r, c))
    if patch:
        O = fill(O, SIX, frozenset(patch))

def dsl4(I):
    # Draw a one-cell outer frame (outbox) around each BLUE object, background-only.
    O = I
    blue_objs = colorfilter(objects(I, True, False, True), TWO)
    patch = set()
    for obj in blue_objs:
        inds = toindices(obj)
        # outbox: ring just outside the bounding box of the object
        ring = outbox(inds)
        for rc in ring:
            if index(O, rc) == SEVEN:
                patch.add(rc)
    if patch:
        O = fill(O, SIX, frozenset(patch))
    return O

def dsl5(I):
    # Corridor fill: turn background segments strictly between two non-background ends into ORANGE,
    # both horizontally and vertically. Then merge.
    H, W = shape(I)
    O = I
    patch = set()
    # Rows
    for r in range(H):
        c = 0
        while c < W:
            if index(I, (r, c)) != SEVEN:
                c += 1
                continue
            start = c
            while c < W and index(I, (r, c)) == SEVEN:
                c += 1
            end = c - 1
            left_ok = (start - 1 >= 0) and (index(I, (r, start - 1)) != SEVEN)
            right_ok = (c < W) and (index(I, (r, c)) != SEVEN)
            if left_ok and right_ok:
                for j in range(start, end + 1):
                    if index(O, (r, j)) == SEVEN:
                        patch.add((r, j))
    # Columns
    for c in range(W):
        r = 0
        while r < H:
            if index(I, (r, c)) != SEVEN:
                r += 1
                continue
            start = r
            while r < H and index(I, (r, c)) == SEVEN:
                r += 1
            end = r - 1
            up_ok = (start - 1 >= 0) and (index(I, (start - 1, c)) != SEVEN)
            down_ok = (r < H) and (index(I, (r, c)) != SEVEN)
            if up_ok and down_ok:
                for i in range(start, end + 1):
                    if index(O, (i, c)) == SEVEN:
                        patch.add((i, c))
    if patch:
        O = fill(O, SIX, frozenset(patch))
    return O
