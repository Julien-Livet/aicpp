def dsl1(I):
    O = dmirror(I)
    return O

def dsl2(I):
    O = compress(I)
    return O

def dsl3(I):
    R = rot180(I)
    O = cellwise(I, R, mostcolor(I))
    return O

def dsl4(I):
    h = height(I)
    w = width(I)
    h2 = divide(h, TWO)
    top = crop(I, astuple(ZERO, ZERO), astuple(h2, w))
    bot = crop(I, astuple(h2, ZERO), astuple(subtract(h, h2), w))
    O = canvas(leastcolor(I), astuple(h, w))
    O = fill(O, mostcolor(top), shift(asindices(top), astuple(ZERO, ZERO)))
    O = fill(O, mostcolor(bot), shift(asindices(bot), astuple(h2, ZERO)))
    return O

def dsl5(I):
    J = dmirror(I)
    h = height(J)
    w = width(J)
    w2 = divide(w, TWO)
    left = crop(J, astuple(ZERO, ZERO), astuple(h, w2))
    right = crop(J, astuple(ZERO, w2), astuple(h, subtract(w, w2)))
    O = canvas(leastcolor(J), astuple(h, w))
    O = fill(O, mostcolor(left), shift(asindices(left), astuple(ZERO, ZERO)))
    O = fill(O, mostcolor(right), shift(asindices(right), astuple(ZERO, w2)))
    return O
