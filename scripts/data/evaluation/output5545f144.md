def dsl1(I):
    # Consensus between the leftmost and rightmost panels (cellwise keep equal cells, else background)
    bg = mostcolor(I)
    allobjs = objects(I, T, F, T)
    vliners = sfilter(allobjs, vline)
    fullheight = matcher(compose(height, rbind(subgrid, I)), height(I))
    vfull = sfilter(vliners, fullheight)
    npanels = add(size(vfull), ONE)
    tiles = vsplit(I, npanels)
    x = first(tiles)
    y = last(tiles)
    O = cellwise(x, y, bg)
    return O

def dsl4(I):
    # Choose the sparsest panel (fewest non-background cells)
    bg = mostcolor(I)
    totalcells = compose(size, asindices)
    bgcount = rbind(colorcount, bg)
    forecount = fork(subtract, totalcells, bgcount)
    allobjs = objects(I, T, F, T)
    vliners = sfilter(allobjs, vline)
    fullheight = matcher(compose(height, rbind(subgrid, I)), height(I))
    vfull = sfilter(vliners, fullheight)
    npanels = add(size(vfull), ONE)
    tiles = vsplit(I, npanels)
    O = argmin(tiles, forecount)
    return O

def dsl5(I):
    # Remove full-height separator columns (paint them as background), then take the rightmost panel
    bg = mostcolor(I)
    allobjs = objects(I, T, F, T)
    vliners = sfilter(allobjs, vline)
    fullheight = matcher(compose(height, rbind(subgrid, I)), height(I))
    vfull = sfilter(vliners, fullheight)
    sepidx = mapply(toindices, vfull)
    cleaned = fill(I, bg, sepidx)
    npanels = add(size(vfull), ONE)
    tiles = vsplit(cleaned, npanels)
    O = last(tiles)
