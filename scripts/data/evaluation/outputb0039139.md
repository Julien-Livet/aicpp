def dsl1(I):
    # Use the region between the first two vertical 1-separators.
    # Recolor: background->second-rightmost color, foreground->rightmost color, then compress.
    h = height(I)
    w = width(I)
    objs = objects(I, T, F, T)
    ones = colorfilter(objs, ONE)
    vones = order(sfilter(ones, vline), leftmost)
    v1 = first(vones)
    v2 = first(remove(v1, vones))
    jL = leftmost(v1)
    jR = leftmost(v2)
    reg = crop(I, astuple(ZERO, increment(jL)), astuple(h, subtract(jR, increment(jL))))
    mask = compress(reg)
    cols = remove(ONE, remove(ZERO, palette(I)))
    fR = compose(rightmost, lbind(ofcolor, I))
    cR = argmax(cols, fR)
    cols2 = remove(cR, cols)
    cA = argmax(cols2, fR)
    idx_all = asindices(mask)
    idx0 = ofcolor(mask, ZERO)
    idx_fg = difference(idx_all, idx0)
    g1 = fill(mask, cA, idx0)
    O = fill(g1, cR, idx_fg)
    return O

def dsl2(I):
    # Concatenate the two rightmost content regions split by vertical 1-separators.
    # Left (penultimate) recolored to second-rightmost color; Right (last) to rightmost color.
    # Then compress result and fill remaining background with the second-rightmost color.
    h = height(I)
    w = width(I)
    objs = objects(I, T, F, T)
    ones = colorfilter(objs, ONE)
    vones = order(sfilter(ones, vline), leftmost)
    vlast = last(vones)
    vpen = last(remove(vlast, vones))
    j1 = leftmost(vpen)
    j2 = leftmost(vlast)
    R1 = crop(I, astuple(ZERO, increment(j1)), astuple(h, subtract(j2, increment(j1))))
    R2 = crop(I, astuple(ZERO, increment(j2)), astuple(h, subtract(w, increment(j2))))
    cols = remove(ONE, remove(ZERO, palette(I)))
    fR = compose(rightmost, lbind(ofcolor, I))
    cR = argmax(cols, fR)
    cols2 = remove(cR, cols)
    cA = argmax(cols2, fR)
    # Recolor nonzero in R1->cA, nonzero in R2->cR
    r1c = compress(R1)
    r2c = compress(R2)
    r1_all = asindices(r1c)
    r1_0 = ofcolor(r1c, ZERO)
    r1_fg = difference(r1_all, r1_0)
    r1r = fill(r1c, cA, r1_fg)
    r2_all = asindices(r2c)
    r2_0 = ofcolor(r2c, ZERO)
    r2_fg = difference(r2_all, r2_0)
    r2r = fill(r2c, cR, r2_fg)
    H = h
    L = compress(hconcat(r1r, r2r))
    # Fill remaining background with cA to make a two-color output
    allL = asindices(L)
    zL = ofcolor(L, ZERO)
    O = fill(L, cA, zL)
    return O

def dsl3(I):
    # Keep only the two rightmost colors; crop to their joint bounding box; fill background with the second-rightmost color.
    cols = remove(ONE, remove(ZERO, palette(I)))
    fR = compose(rightmost, lbind(ofcolor, I))
    cR = argmax(cols, fR)
    cols2 = remove(cR, cols)
    cA = argmax(cols2, fR)
    keep = combine(ofcolor(I, cA), ofcolor(I, cR))
    cropped = subgrid(keep, I)
    base = compress(cropped)
    allB = asindices(base)
    zB = ofcolor(base, ZERO)
    O = fill(base, cA, zB)
    return O

def dsl5(I):
    # If vertical 1-separators exist, use the middle mask as in dsl1 but horizontally upscaled.
    # Else (fallback) use the region between first two horizontal 1-separators similarly.
    cols = remove(ONE, remove(ZERO, palette(I)))
    fR = compose(rightmost, lbind(ofcolor, I))
    cR = argmax(cols, fR)
    cols2 = remove(cR, cols)
    cA = argmax(cols2, fR)
    h = height(I)
    w = width(I)
    objs = objects(I, T, F, T)
    ones = colorfilter(objs, ONE)
    vmask = sfilter(ones, vline)
    cond = greater(size(vmask), ZERO)
    def do_vert(I):
        vs = order(vmask, leftmost)
        v1 = first(vs)
        v2 = first(remove(v1, vs))
        jL = leftmost(v1)
        jR = leftmost(v2)
        reg = crop(I, astuple(ZERO, increment(jL)), astuple(h, subtract(jR, increment(jL))))
        return compress(reg)
    def do_horz(I):
        hmask = order(sfilter(ones, hline), uppermost)
        h1 = first(hmask)
        h2 = first(remove(h1, hmask))
        iT = uppermost(h1)
        iB = uppermost(h2)
        reg = crop(I, astuple(increment(iT), ZERO), astuple(subtract(iB, increment(iT)), w))
        return compress(reg)
    mask = branch(cond, do_vert(I), do_horz(I))
    # Recolor and mildly upscale horizontally to emphasize runs
    idx_all = asindices(mask)
    idx0 = ofcolor(mask, ZERO)
    idx_fg = difference(idx_all, idx0)
    g1 = fill(mask, cA, idx0)
    g2 = fill(g1, cR, idx_fg)
    O = hupscale(g2, THREE)
    return O
