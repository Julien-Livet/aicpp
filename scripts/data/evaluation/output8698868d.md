def dsl1(I):
    # Remove redundant frontiers globally
    O = compress(I)
    return O

def dsl2(I):
    # Compress each vertical half separately, then stitch
    L = lefthalf(I)
    R = righthalf(I)
    LC = compress(L)
    RC = compress(R)
    O = hconcat(LC, RC)
    return O

def dsl3(I):
    # Compress each horizontal half separately, then stitch
    T = tophalf(I)
    B = bottomhalf(I)
    TC = compress(T)
    BC = compress(B)
    O = vconcat(TC, BC)
    return O

def dsl4(I):
    # Orientation-robust compress (rotate, compress, rotate back)
    x = rot90(I)
    y = compress(x)
    O = rot270(y)
    return O

def dsl5(I):
    # Compress each quadrant (2x2 split) and reassemble
    top, bottom = hsplit(I, 2)
    tl, tr = vsplit(top, 2)
    bl, br = vsplit(bottom, 2)
    TL = compress(tl)
    TR = compress(tr)
    BL = compress(bl)
    BR = compress(br)
    top_row = hconcat(TL, TR)
    bot_row = hconcat(BL, BR)
    O = vconcat(top_row, bot_row)
    return O
