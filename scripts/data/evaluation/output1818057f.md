def dsl1(I):
    # Recolor all plus-shaped 4s (center and its 4-neighbors) to 8 using set-shift intersections
    S = ofcolor(I, FOUR)
    a = shift(S, UP)
    b = shift(S, DOWN)
    c = shift(S, LEFT)
    d = shift(S, RIGHT)
    t1 = intersection(S, a)
    t2 = intersection(t1, b)
    t3 = intersection(t2, c)
    centers = intersection(t3, d)
    upc = shift(centers, UP)
    downc = shift(centers, DOWN)
    leftc = shift(centers, LEFT)
    rightc = shift(centers, RIGHT)
    p1 = combine(centers, upc)
    p2 = combine(p1, downc)
    p3 = combine(p2, leftc)
    pluscells = combine(p3, rightc)
    O = fill(I, EIGHT, pluscells)
    return O

def dsl2(I):
    # Find plus object of 4s via occurrences and paint the same shape recolored to 8
    p = initset(ORIGIN)
    p = insert(UP, p)
    p = insert(DOWN, p)
    p = insert(LEFT, p)
    p = insert(RIGHT, p)
    plus4 = recolor(FOUR, p)
    locs = occurrences(I, plus4)
    plus8 = recolor(EIGHT, p)
    shifter = lbind(shift, plus8)
    objs = mapply(shifter, locs)
    O = paint(I, objs)
    return O

def dsl4(I):
    # Detect centers via separate horizontal and vertical triple-4 conditions then expand to plus and recolor
    S = ofcolor(I, FOUR)
    h1 = intersection(S, shift(S, LEFT))
    hcenters = intersection(h1, shift(S, RIGHT))
    v1 = intersection(S, shift(S, UP))
    vcenters = intersection(v1, shift(S, DOWN))
    centers = intersection(hcenters, vcenters)
    upc = shift(centers, UP)
    downc = shift(centers, DOWN)
    leftc = shift(centers, LEFT)
    rightc = shift(centers, RIGHT)
    u1 = combine(centers, upc)
    u2 = combine(u1, downc)
    u3 = combine(u2, leftc)
    pluscells = combine(u3, rightc)

    return O

def dsl5(I):
    # Occurrence-based: place 8-colored pluses using prapply over a singleton plus and all occurrence offsets
    p = initset(ORIGIN)
    p = insert(UP, p)
    p = insert(DOWN, p)
    p = insert(LEFT, p)
    p = insert(RIGHT, p)
    plus4 = recolor(FOUR, p)
    locs = occurrences(I, plus4)
    plus8 = recolor(EIGHT, p)
    single = initset(plus8)
    shifted_objs = prapply(shift, single, locs)
    objs = merge(shifted_objs)
    O = paint(I, objs)
