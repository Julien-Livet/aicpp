def dsl1(I):
    parts = objects(I, T, F, T)
    six = colorfilter(parts, SIX)
    base = replace(I, SIX, SEVEN)
    s2 = sizefilter(six, TWO)
    h2 = mfilter(s2, hline)
    v2 = mfilter(s2, vline)
    s3 = sizefilter(six, THREE)
    h3 = mfilter(s3, hline)
    v3 = mfilter(s3, vline)
    line3 = combine(h3, v3)
    l3 = difference(s3, line3)
    s4 = sizefilter(six, FOUR)
    s5 = sizefilter(six, FIVE)
    s6 = sizefilter(six, SIX)
    o8h2 = mapply(lbind(recolor, EIGHT), h2)
    o9v2 = mapply(lbind(recolor, NINE), v2)
    o2line3 = mapply(lbind(recolor, TWO), line3)
    o4l3 = mapply(lbind(recolor, FOUR), l3)
    o8s4 = mapply(lbind(recolor, EIGHT), s4)
    o3s5 = mapply(lbind(recolor, THREE), s5)
    o5s6 = mapply(lbind(recolor, FIVE), s6)
    recol = combine(combine(combine(o8h2, o9v2), combine(o2line3, o4l3)), combine(o8s4, combine(o3s5, o5s6)))
    O = paint(base, recol)
    return O

def dsl5(I):
    parts = objects(I, T, F, T)
    six = colorfilter(parts, SIX)
    base = replace(I, SIX, SEVEN)
    # classify first by lines vs non-lines, then by size
    lines = combine(mfilter(six, hline), mfilter(six, vline))
    nonlines = difference(six, lines)
    s2 = sizefilter(lines, TWO)
    h2 = mfilter(s2, hline)
    v2 = mfilter(s2, vline)
    s3line = sizefilter(lines, THREE)
    s3non = sizefilter(nonlines, THREE)
    s4 = sizefilter(nonlines, FOUR)
    s5 = sizefilter(nonlines, FIVE)
    s6 = sizefilter(nonlines, SIX)
    o8h2 = mapply(lbind(recolor, EIGHT), h2)

    o4_3non = mapply(lbind(recolor, FOUR), s3non)
    o8_4 = mapply(lbind(recolor, EIGHT), s4)
    o3_5 = mapply(lbind(recolor, THREE), s5)
    o5_6 = mapply(lbind(recolor, FIVE), s6)
