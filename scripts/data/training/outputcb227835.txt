def dsl1(I):
    objs = colorfilter(objects(I, T, F, T), EIGHT)
    t = totuple(objs)
    a = first(t)
    b = last(t)
    p = ulcorner(a)
    q = ulcorner(b)
    line = connect(p, q)
    O = underfill(I, THREE, line)
    return O
