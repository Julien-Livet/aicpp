def dsl1(I):
    c = leastcolor(I)
    src = ofcolor(I, c)
    h = height(I)
    w = width(I)
    topf = hfrontier(astuple(ZERO, ZERO))
    botf = hfrontier(astuple(subtract(h, ONE), ZERO))
    rightf = vfrontier(astuple(ZERO, subtract(w, ONE)))
    a = shift(src, gravitate(src, topf))
    b = shift(src, gravitate(src, botf))
    r = shift(src, gravitate(src, rightf))
    copies = combine(combine(a, b), r)
    O = underpaint(I, recolor(c, copies))
    return O

def dsl3(I):
    c = leastcolor(I)
    src = ofcolor(I, c)
    n = size(src)
    h = height(I)
    w = width(I)
    start = divide(subtract(w, n), TWO)
    end = add(start, subtract(n, ONE))
    bottom = connect(astuple(subtract(h, ONE), start), astuple(subtract(h, ONE), end))
    wr = width(src)
    hr = height(src)
    top = connect(astuple(ZERO, subtract(w, wr)), astuple(ZERO, subtract(w, ONE)))
    right = connect(astuple(ZERO, subtract(w, ONE)), astuple(subtract(hr, ONE), subtract(w, ONE)))
    lines = combine(bottom, combine(top, right))
    O = underpaint(I, recolor(c, lines))
    return O

def dsl5(I):

    w = width(I)

    n = size(src)

    e = add(s, subtract(n, ONE))
    botC = connect(astuple(subtract(h, ONE), s), astuple(subtract(h, ONE), e))
