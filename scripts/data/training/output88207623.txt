def dsl1(I):
    # per-half: align vmirrored 4s to 2-column using centers; paint with most frequent seed color per half
    h = height(I)
    th = divide(h, 2)
    T = tophalf(I)
    B = bottomhalf(I)

    # top half
    S2t = ofcolor(T, TWO)
    S2tm = ofcolor(vmirror(T), TWO)
    oj_t = subtract(center(S2t), center(S2tm))
    Mt = move(canvas(ZERO, shape(T)), recolor(ONE, ofcolor(vmirror(T), FOUR)), oj_t)
    St = ofcolor(Mt, ONE)
    ct = mostcolor(replace(replace(replace(T, ZERO, TEN), TWO, TEN), FOUR, TEN))

    # bottom half
    S2b = ofcolor(B, TWO)
    S2bm = ofcolor(vmirror(B), TWO)
    oj_b = subtract(center(S2b), center(S2bm))
    Mb = move(canvas(ZERO, shape(B)), recolor(ONE, ofcolor(vmirror(B), FOUR)), oj_b)
    Sb = ofcolor(Mb, ONE)
    cb = mostcolor(replace(replace(replace(B, ZERO, TEN), TWO, TEN), FOUR, TEN))

    G = underpaint(I, recolor(ct, St))
    O = underpaint(G, recolor(cb, shift(Sb, astuple(th, 0))))
    return O
