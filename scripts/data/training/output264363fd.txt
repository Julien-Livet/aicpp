def dsl1(I):
    # Draw row/column lines through each 4: use 3 over 2-cells and 2 over 1-cells
    seeds4 = ofcolor(I, FOUR)
    H = merge(apply(hfrontier, seeds4))
    V = merge(apply(vfrontier, seeds4))
    HV = combine(H, V)
    on1 = intersection(HV, ofcolor(I, ONE))
    on2 = intersection(HV, ofcolor(I, TWO))
    O = fill(I, TWO, on1)
    O = fill(O, THREE, on2)
    O = fill(O, FOUR, seeds4)
    return O

def dsl2(I):
    # Around each 4: paint onto 3-background only (diagonals -> 5, orthogonals -> 6)
    seeds4 = ofcolor(I, FOUR)
    diag = merge(apply(ineighbors, seeds4))
    ortho = merge(apply(neighbors, seeds4))
    on3d = intersection(diag, ofcolor(I, THREE))
    on3o = intersection(ortho, ofcolor(I, THREE))
    O = fill(I, FIVE, on3d)
    O = fill(O, SIX, on3o)
    O = fill(O, FOUR, seeds4)
    return O

def dsl3(I):
    # Extend every existing 3 into full row/column lines over 1/2 background
    seeds3 = ofcolor(I, THREE)
    H = merge(apply(hfrontier, seeds3))
    V = merge(apply(vfrontier, seeds3))
    HV = combine(H, V)
    bg12 = combine(ofcolor(I, ONE), ofcolor(I, TWO))
    draw = intersection(HV, bg12)
    O = fill(I, THREE, draw)
    return O

def dsl4(I):
    # Turn only the boundary cells of 2-regions into 3 (2-cells that lie on a frontier)
    F = merge(frontiers(I))
    boundary2 = intersection(F, ofcolor(I, TWO))
    O = fill(I, THREE, boundary2)
    return O

def dsl5(I):
    # Local cross around each 4 over 1/2 background using 3
    seeds4 = ofcolor(I, FOUR)
    ortho = merge(apply(neighbors, seeds4))
    diag = merge(apply(ineighbors, seeds4))
    cross = combine(ortho, diag)
    bg12 = combine(ofcolor(I, ONE), ofcolor(I, TWO))
    mask = intersection(cross, bg12)
    O = fill(I, THREE, mask)
    O = fill(O, FOUR, seeds4)
    return O
