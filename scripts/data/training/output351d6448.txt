def dsl1(I):
    # Take the first and last block-objects (ignoring 5s). If they have equal width, use just one; else concatenate them; center on middle row.
    x1 = replace(I, FIVE, ZERO)
    objs = objects(x1, F, F, T)
    W = width(I)
    t = totuple(objs)
    rows = order(apply(uppermost, t), identity)
    r0 = first(rows)
    rk = last(rows)
    o0 = merge(sfilter(objs, matcher(uppermost, r0)))
    ok = merge(sfilter(objs, matcher(uppermost, rk)))
    w0 = width(o0)
    wk = width(ok)
    same = equality(w0, wk)
    # Build small piece
    small_w = branch(same, w0, add(w0, wk))
    small = canvas(ZERO, astuple(3, small_w))
    p0 = shift(normalize(o0), astuple(1, 0))
    small = paint(small, p0)
    small = branch(same, small, paint(small, shift(normalize(ok), astuple(1, w0))))
    # Center on full width
    joff = divide(subtract(W, small_w), TWO)
    o_small = asobject(small)
    O = paint(canvas(ZERO, astuple(3, W)), shift(normalize(o_small), astuple(1, joff)))
    return O

def dsl2(I):
    # Concatenate the first (topmost) and last (bottommost) block-objects (ignoring 5s) and left-align on middle row.
    x1 = replace(I, FIVE, ZERO)
    objs = objects(x1, F, F, T)
    t = totuple(objs)
    rows = order(apply(uppermost, t), identity)
    r0 = first(rows)
    rk = last(rows)
    o0 = merge(sfilter(objs, matcher(uppermost, r0)))
    ok = merge(sfilter(objs, matcher(uppermost, rk)))
    w0 = width(o0)
    wk = width(ok)
    small = canvas(ZERO, astuple(3, add(w0, wk)))
    p0 = shift(normalize(o0), astuple(1, 0))
    pk = shift(normalize(ok), astuple(1, w0))
    small = paint(paint(small, p0), pk)
    W = width(I)
    O = paint(canvas(ZERO, astuple(3, W)), shift(asobject(small), astuple(0, 0)))
    return O

def dsl3(I):
    # Overlay-union of all block-objects normalized to j=0, then center the union on the middle row.
    x1 = replace(I, FIVE, ZERO)
    objs = objects(x1, F, F, T)
    # Build a small canvas as wide as the maximum width
    omax = argmax(objs, width)
    wM = width(omax)
    small = canvas(ZERO, astuple(3, wM))
    # Paint all objects normalized at j=0 on the middle row
    t = totuple(objs)
    def place(o):
        return shift(normalize(o), astuple(1, 0))
    ps = apply(place, t)
    small2 = merge(apply(lbind(paint, small), ps))
    small2 = branch(equality(small2, ZERO), small, small2)
    W = width(I)
    joff = divide(subtract(W, wM), TWO)
    O = paint(canvas(ZERO, astuple(3, W)), shift(asobject(small2), astuple(0, joff)))
    return O

def dsl4(I):
    # Concatenate all block-objects (top to bottom order) and center them on the middle row.
    x1 = replace(I, FIVE, ZERO)
    objs = objects(x1, F, F, T)
    t = totuple(objs)
    ordt = order(t, uppermost)
    # Compute total width
    ws = apply(width, ordt)
    # Accumulate paints left-to-right
    total_w = add(add(add(0, first(ws)), branch(greater(size(ordt), ONE), index(ws, astuple(1, 0)), 0)),
                  branch(greater(size(ordt), TWO), index(ws, astuple(2, 0)), 0))
    total_w = add(total_w, branch(greater(size(ordt), THREE), index(ws, astuple(3, 0)), 0))
    # Fallback if more than 4: just sum widths by iteratively adding width of arg
    # Build canvas with conservative width: width of I
    W = width(I)
    small = canvas(ZERO, astuple(3, W))
    # Place sequentially until width
    off0 = 0
    canv = small
    def place_acc(canv_o_pair):
        canv_cur = first(canv_o_pair)
        o = last(canv_o_pair)
        w = width(o)
        obj = shift(normalize(o), astuple(1, width(asobject(canv_cur))))
        return paint(canv_cur, obj)
    # Paint each ordered object shifted by cumulative width (approximated by repeated painting at current asobject width)
    canv = paint(canv, shift(normalize(first(ordt)), astuple(1, 0)))
    canv = branch(greater(size(ordt), ONE), paint(canv, shift(normalize(index(ordt, astuple(1, 0))), astuple(1, width(asobject(canv))))), canv)
    canv = branch(greater(size(ordt), TWO), paint(canv, shift(normalize(index(ordt, astuple(2, 0))), astuple(1, width(asobject(canv))))), canv)
    canv = branch(greater(size(ordt), THREE), paint(canv, shift(normalize(index(ordt, astuple(3, 0))), astuple(1, width(asobject(canv))))), canv)
    # Center result object within W
    o_small = asobject(canv)
    ww = width(o_small)
    joff = divide(subtract(W, ww), TWO)
    O = paint(canvas(ZERO, astuple(3, W)), shift(normalize(o_small), astuple(1, joff)))
    return O

def dsl5(I):
    # Choose the median-width block-object (second smallest when multiple), and center it on the middle row.
    x1 = replace(I, FIVE, ZERO)
    objs = objects(x1, F, F, T)
    t = totuple(objs)
    # Sort by width
    ordw = order(t, width)
    only = first(ordw)
    lasto = last(ordw)
    inner = remove(only, remove(lasto, ordw))
    med = branch(greater(size(objs), TWO), first(inner), lasto)
    o = merge(sfilter(objs, matcher(identity, med)))
    W = width(I)
    w = width(o)
    joff = divide(subtract(W, w), TWO)
    O = paint(canvas(ZERO, astuple(3, W)), shift(normalize(o), astuple(1, joff)))
    return O
