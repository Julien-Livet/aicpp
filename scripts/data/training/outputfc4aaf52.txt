def dsl1(I):
    fg = difference(asindices(I), ofcolor(I, EIGHT))
    S = subgrid(fg, I)
    h = height(S)
    w = width(S)
    th = divide(h, 2)
    top = crop(S, ORIGIN, astuple(th, w))
    bot = crop(S, astuple(th, 0), astuple(subtract(h, th), w))
    cols = remove(EIGHT, palette(S))
    a = minimum(cols)
    b = maximum(cols)
    top2 = switch(top, a, b)
    bot2 = switch(bot, a, b)
    top_idx = difference(asindices(top2), ofcolor(top2, EIGHT))
    bot_idx = difference(asindices(bot2), ofcolor(bot2, EIGHT))
    top_obj = toobject(top_idx, top2)
    bot_obj = toobject(bot_idx, bot2)
    O = canvas(EIGHT, shape(I))
    ul = ulcorner(fg)
    O1 = paint(O, shift(top_obj, add(ul, tojvec(w))))
    O2 = paint(O1, shift(bot_obj, add(ul, toivec(th))))
    return O2

def dsl4(I):
    all_nonbg = difference(asindices(I), ofcolor(I, EIGHT))
    bbox_ul = ulcorner(all_nonbg)
    S = subgrid(all_nonbg, I)
    h = height(S)
    w = width(S)
    th = divide(h, 2)
    top = crop(S, ORIGIN, astuple(th, w))
    bot = crop(S, astuple(th, 0), astuple(subtract(h, th), w))
    # robust color swap using detected two colors
    cols = remove(EIGHT, palette(S))
    c_lo = minimum(cols)
    c_hi = maximum(cols)
    top_sw = switch(top, c_lo, c_hi)
    bot_sw = switch(bot, c_lo, c_hi)
    # convert to non-background objects
    t_idx = difference(asindices(top_sw), ofcolor(top_sw, EIGHT))
    b_idx = difference(asindices(bot_sw), ofcolor(bot_sw, EIGHT))
    t_obj = toobject(t_idx, top_sw)
    b_obj = toobject(b_idx, bot_sw)
    O = canvas(EIGHT, shape(I))
    O = paint(O, shift(t_obj, add(bbox_ul, tojvec(w))))
    O = paint(O, shift(b_obj, add(bbox_ul, toivec(th))))
    return O

    S = subgrid(fg, I)

    top = crop(S, ORIGIN, astuple(th, w))
    bot = crop(S, astuple(th, 0), astuple(subtract(h, th), w))
    # determine colors via most/least to avoid set ordering

    top2 = switch(top, a, b)
    bot2 = switch(bot, a, b)
    t_obj = toobject(difference(asindices(top2), ofcolor(top2, EIGHT)), top2)
    b_obj = toobject(difference(asindices(bot2), ofcolor(bot2, EIGHT)), bot2)
    O = canvas(EIGHT, shape(I))
    ul = ulcorner(fg)
    O = paint(O, shift(t_obj, add(ul, tojvec(w))))
    O = paint(O, shift(b_obj, add(ul, toivec(th))))
    return O
