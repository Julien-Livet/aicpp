def dsl1(I):
    five = ofcolor(I, FIVE)
    H = height(five)
    W = width(five)
    dims = astuple(H, W)
    base = canvas(ZERO, dims)
    allobjs = fgpartition(I)
    nozero = difference(allobjs, colorfilter(allobjs, ZERO))
    objs = difference(nozero, colorfilter(allobjs, FIVE))
    vset = mfilter(objs, vline)
    hset = mfilter(objs, hline)
    vb = size(vset)
    hb = size(hset)
    rowobjs = order(objs, uppermost)
    rowcolors = apply(color, rowobjs)
    is_ = interval(0, H, 1)
    fR = rbind(shoot, RIGHT)
    gR = compose(fR, toivec)
    rowpatches = apply(gR, is_)
    recolR = papply(recolor, rowcolors, rowpatches)
    objR = merge(recolR)
    Or = paint(base, objR)
    colobjs = order(objs, leftmost)
    colcolors = apply(color, colobjs)
    js = interval(0, W, 1)
    fC = rbind(shoot, DOWN)
    gC = compose(fC, tojvec)
    colpatches = apply(gC, js)
    recolC = papply(recolor, colcolors, colpatches)
    objC = merge(recolC)
    Oc = paint(base, objC)
    useCols = greater(vb, hb)
    O = branch(useCols, Oc, Or)
    return O

def dsl5(I):
    five = ofcolor(I, FIVE)
    H = height(five)
    W = width(five)
    base = canvas(ZERO, astuple(H, W))
    allobjs = fgpartition(I)
    nz = difference(allobjs, colorfilter(allobjs, ZERO))
    objs = difference(nz, colorfilter(allobjs, FIVE))
    # prefer columns when vertical lines dominate, else rows
    vset = mfilter(objs, vline)
    hset = mfilter(objs, hline)
    preferCols = greater(size(vset), size(hset))
    # build rows version
    rObjs = order(objs, uppermost)
    rCols = apply(color, rObjs)
    is_ = interval(0, H, 1)
    rFun = compose(rbind(shoot, RIGHT), toivec)
    rPatches = apply(rFun, is_)
    rRecol = papply(recolor, rCols, rPatches)
    rObj = merge(rRecol)
    Or = paint(base, rObj)
    # build cols version
    cObjs = order(objs, leftmost)
    cCols = apply(color, cObjs)
    js = interval(0, W, 1)
    cFun = compose(rbind(shoot, DOWN), tojvec)
    cPatches = apply(cFun, js)
    cRecol = papply(recolor, cCols, cPatches)
    cObj = merge(cRecol)
    Oc = paint(base, cObj)
    O = branch(preferCols, Oc, Or)
    return O
