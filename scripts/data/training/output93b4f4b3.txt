def dsl1(I):
    L = lefthalf(I)
    R = righthalf(I)
    b = mostcolor(L)
    base = replace(L, ZERO, b)
    zs = colorfilter(objects(L, T, F, T), ZERO)
    rs = objects(R, T, F, T)
    za = argmin(zs, uppermost)
    zc = argmax(zs, uppermost)
    zb = first(remove(za, remove(zc, zs)))
    sza = toindices(normalize(za))
    ca = fork(equality, compose(toindices, normalize), rbind(identity, sza))
    ra = extract(rs, ca)
    va = subtract(ulcorner(za), ulcorner(ra))
    sa = shift(ra, va)
    szb = toindices(normalize(zb))
    cb = fork(equality, compose(toindices, normalize), rbind(identity, szb))
    rb = extract(rs, cb)
    vb = subtract(ulcorner(zb), ulcorner(rb))
    sb = shift(rb, vb)
    szc = toindices(normalize(zc))
    cc = fork(equality, compose(toindices, normalize), rbind(identity, szc))
    rc = extract(rs, cc)
    vc = subtract(ulcorner(zc), ulcorner(rc))
    sc = shift(rc, vc)
    S = insert(sa, initset(sc))
    S = insert(sb, S)
    M = merge(S)
    O = paint(base, M)
    return O

def dsl3(I):
    L = lefthalf(I)
    R = righthalf(I)
    b = mostcolor(L)
    base = replace(L, ZERO, b)
    holes = colorfilter(objects(L, T, F, T), ZERO)
    rs = objects(R, T, F, T)
    top = argmin(holes, uppermost)
    mid = first(remove(top, remove(argmax(holes, uppermost), holes)))
    bot = argmax(holes, uppermost)
    forA = toindices(normalize(top))
    condA = fork(equality, compose(toindices, normalize), rbind(identity, forA))
    rA = extract(rs, condA)
    offA = subtract(ulcorner(top), ulcorner(rA))
    sA = shift(rA, offA)
    forB = toindices(normalize(mid))
    condB = fork(equality, compose(toindices, normalize), rbind(identity, forB))
    rB = extract(rs, condB)
    offB = subtract(ulcorner(mid), ulcorner(rB))
    sB = shift(rB, offB)
    forC = toindices(normalize(bot))
    condC = fork(equality, compose(toindices, normalize), rbind(identity, forC))
    rC = extract(rs, condC)
    offC = subtract(ulcorner(bot), ulcorner(rC))
    sC = shift(rC, offC)
    pack = insert(sA, initset(sB))
    pack = insert(sC, pack)
    moved = merge(pack)
    O = paint(base, moved)
    return O
