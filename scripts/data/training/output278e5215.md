def dsl1(I):
    # Robust: find bbox of 'on' object (color FIVE), extract 3-row palette by trimming the covered grid,
    # pick the duplicated non-uniform row as foreground and the uniform row as background.
    p5 = ofcolor(I, FIVE)
    s5 = subgrid(p5, I)
    H = height(s5)
    W = width(s5)
    base = cover(I, toobject(p5, I))
    pal = trim(base)
    r0 = crop(pal, ORIGIN, astuple(ONE, W))
    r1 = crop(pal, astuple(ONE, ZERO), astuple(ONE, W))
    r2 = crop(pal, astuple(TWO, ZERO), astuple(ONE, W))
    v0 = index(r0, ORIGIN)
    v1 = index(r1, ORIGIN)
    v2 = index(r2, ORIGIN)
    u0 = equality(r0, canvas(v0, astuple(ONE, W)))
    u1 = equality(r1, canvas(v1, astuple(ONE, W)))
    u2 = equality(r2, canvas(v2, astuple(ONE, W)))
    e01 = equality(r0, r1)
    e02 = equality(r0, r2)
    e12 = equality(r1, r2)
    # Foreground pattern row = duplicated non-uniform row (fallback: first non-uniform)
    prow = branch(e01, r0, branch(e02, r0, branch(e12, r1, branch(flip(u0), r0, branch(flip(u1), r1, r2)))))
    # Background row = the uniform row (fallback: r2)
    brow = branch(u0, r0, branch(u1, r1, branch(u2, r2, r2)))
    Agrid = vupscale(prow, H)
    Bgrid = vupscale(brow, H)
    onmask = ofcolor(s5, FIVE)
    O = paint(Bgrid, toobject(onmask, Agrid))
    return O

def dsl2(I):
    # Majority-by-equality first, otherwise pick most non-uniform as foreground and uniform as background.
    p5 = ofcolor(I, FIVE)
    s5 = subgrid(p5, I)
    H = height(s5)
    W = width(s5)
    pal = trim(cover(I, toobject(p5, I)))
    r0 = crop(pal, ORIGIN, astuple(ONE, W))
    r1 = crop(pal, astuple(ONE, ZERO), astuple(ONE, W))
    r2 = crop(pal, astuple(TWO, ZERO), astuple(ONE, W))
    v0 = index(r0, ORIGIN); u0 = equality(r0, canvas(v0, astuple(ONE, W)))
    v1 = index(r1, ORIGIN); u1 = equality(r1, canvas(v1, astuple(ONE, W)))
    v2 = index(r2, ORIGIN); u2 = equality(r2, canvas(v2, astuple(ONE, W)))
    e01 = equality(r0, r1); e02 = equality(r0, r2); e12 = equality(r1, r2)
    prow_guess = branch(e01, r0, branch(e02, r0, branch(e12, r1, r0)))
    # If the guessed row is uniform, switch to a non-uniform fallback order r0->r1->r2
    prow = branch(flip(u0), r0, branch(flip(u1), r1, branch(flip(u2), r2, prow_guess)))
    brow = branch(u0, r0, branch(u1, r1, r2))
    Agrid = vupscale(prow, H)
    Bgrid = vupscale(brow, H)
    onmask = ofcolor(s5, FIVE)
    O = paint(Bgrid, toobject(onmask, Agrid))
    return O

def dsl4(I):
    # Column-wise majority palette (two identical rows win), with uniform row as background.
    p5 = ofcolor(I, FIVE)
    s5 = subgrid(p5, I)
    H = height(s5); W = width(s5)
    pal = trim(cover(I, toobject(p5, I)))
    r0 = crop(pal, ORIGIN, astuple(ONE, W))
    r1 = crop(pal, astuple(ONE, ZERO), astuple(ONE, W))
    r2 = crop(pal, astuple(TWO, ZERO), astuple(ONE, W))
    # Determine background as the unique uniform row
    v0 = index(r0, ORIGIN); u0 = equality(r0, canvas(v0, astuple(ONE, W)))
    v1 = index(r1, ORIGIN); u1 = equality(r1, canvas(v1, astuple(ONE, W)))
    v2 = index(r2, ORIGIN); u2 = equality(r2, canvas(v2, astuple(ONE, W)))
    brow = branch(u0, r0, branch(u1, r1, r2))
    # If r0==r1 or r0==r2, prefer r0 as the column-wise winner; else if r1==r2, prefer r1; else fallback to r0
    e01 = equality(r0, r1); e02 = equality(r0, r2); e12 = equality(r1, r2)
    prow = branch(e01, r0, branch(e02, r0, branch(e12, r1, r0)))
    Agrid = vupscale(prow, H)
    Bgrid = vupscale(brow, H)
    onmask = ofcolor(s5, FIVE)
    O = paint(Bgrid, toobject(onmask, Agrid))
    return O

def dsl5(I):

    p5 = ofcolor(I, FIVE)
