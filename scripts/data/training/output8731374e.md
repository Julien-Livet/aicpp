def dsl1(I):
    # Downscale by 2, then draw a centered cross using least vs most common colors
    A = downscale(I, TWO)
    bg = mostcolor(A)
    fg = leastcolor(A)
    sh = shape(A)
    h = first(sh)
    w = last(sh)
    h1 = decrement(h)
    w1 = decrement(w)
    r = halve(h)
    c = halve(w)
    p1 = astuple(r, ZERO)
    p2 = astuple(r, w1)
    q1 = astuple(ZERO, c)
    q2 = astuple(h1, c)
    rowline = connect(p1, p2)
    colline = connect(q1, q2)
    lines = combine(rowline, colline)
    C = canvas(bg, sh)
    O = fill(C, fg, lines)
    return O

def dsl2(I):
    # Extract frontiers, binarize to least/most color, then downscale by 3
    bg = mostcolor(I)
    fg = leastcolor(I)
    sh = shape(I)
    C = canvas(bg, sh)
    Fs = frontiers(I)
    idx = mapply(toindices, Fs)
    G = fill(C, fg, idx)
    O = downscale(G, THREE)
    return O

def dsl3(I):
    # Take the bbox of the largest component of the most common color and draw a centered cross in that box
    bgc = mostcolor(I)
    OBJS = objects(I, T, F, F)
    BGs = colorfilter(OBJS, bgc)
    bigBG = argmax(BGs, size)
    B = subgrid(bigBG, I)
    sh = shape(B)
    h = first(sh)
    w = last(sh)
    h1 = decrement(h)
    w1 = decrement(w)

def dsl4(I):
    # Binarize the top-left quadrant (top half of left half) to least/most common colors
    A = lefthalf(tophalf(I))
    bg = mostcolor(A)
    fg = leastcolor(A)
    sh = shape(A)
    allidx = asindices(A)
    bgidx = ofcolor(A, bg)
    notbg = difference(allidx, bgidx)
    C = canvas(bg, sh)
    O = fill(C, fg, notbg)
    return O

def dsl5(I):
    # Downscale by 3, then draw border columns, a center column, and a center row (striped cross motif)
    A = downscale(I, THREE)
    bg = mostcolor(A)
    fg = leastcolor(A)
    sh = shape(A)
    h = first(sh)
    w = last(sh)
    h1 = decrement(h)
    w1 = decrement(w)
    r = halve(h)
    c = halve(w)
    # vertical lines: left border, center, right border
    vL = connect(astuple(ZERO, ZERO), astuple(h1, ZERO))
    vC = connect(astuple(ZERO, c), astuple(h1, c))
    vR = connect(astuple(ZERO, w1), astuple(h1, w1))
    # horizontal center line
    hC = connect(astuple(r, ZERO), astuple(r, w1))
    lines1 = combine(vL, vC)
    lines2 = combine(vR, hC)
    lines = combine(lines1, lines2)
    C = canvas(bg, sh)
    O = fill(C, fg, lines)
    return O
