def dsl1(I):
    objs = objects(I, T, F, T)
    base = argmax(objs, fork(multiply, height, width))
    basecol = color(base)
    cols = remove(ZERO, palette(I))
    minor = other(cols, basecol)
    dul = add(UP, LEFT)
    dur = add(UP, RIGHT)
    dll = add(DOWN, LEFT)
    drr = add(DOWN, RIGHT)
    ul = ulcorner(base)
    ur = urcorner(base)
    ll = llcorner(base)
    lr = lrcorner(base)
    s_ul = add(ul, dul)
    s_ur = add(ur, dur)
    s_ll = add(ll, dll)
    s_lr = add(lr, drr)
    A = asindices(I)
    r_ul = intersection(A, shoot(s_ul, dul))
    r_ur = intersection(A, shoot(s_ur, dur))
    r_ll = intersection(A, shoot(s_ll, dll))
    r_lr = intersection(A, shoot(s_lr, drr))
    rays = combine(combine(r_ul, r_ur), combine(r_ll, r_lr))
    obj = recolor(minor, rays)
    O = underpaint(I, obj)
    return O

def dsl3(I):
    # Paint each corner ray sequentially
    objs = objects(I, T, F, T)
    base = argmax(objs, fork(multiply, height, width))
    basecol = color(base)
    cols = remove(ZERO, palette(I))
    minor = other(cols, basecol)
    dul = add(UP, LEFT)
    dur = add(UP, RIGHT)
    dll = add(DOWN, LEFT)
    drr = add(DOWN, RIGHT)
    ul = ulcorner(base)
    ur = urcorner(base)
    ll = llcorner(base)
    lr = lrcorner(base)
    A = asindices(I)
    O = I
    for_start = ( (add(ul, dul), dul), (add(ur, dur), dur), (add(ll, dll), dll), (add(lr, drr), drr) )
    # Unroll the loop manually using primitives
    s1 = first(for_start)
    p1 = intersection(A, shoot(first(s1), last(s1)))
    O = underpaint(O, recolor(minor, p1))
    s2 = last(remove(s1, for_start))  # extract next pair in a primitive-compatible way is not directly possible; unroll explicitly instead
    s_ul = add(ul, dul)
    s_ur = add(ur, dur)
    s_ll = add(ll, dll)
    s_lr = add(lr, drr)
    p2 = intersection(A, shoot(s_ur, dur))
    O = underpaint(O, recolor(minor, p2))
    p3 = intersection(A, shoot(s_ll, dll))
    O = underpaint(O, recolor(minor, p3))
    p4 = intersection(A, shoot(s_lr, drr))
    O = underpaint(O, recolor(minor, p4))
    return O

def dsl4(I):

    objs = objects(I, T, F, T)

    ul = ulcorner(base)

def dsl5(I):
    # Using backdrop of base to derive corners (equivalent to bounding box)
    objs = objects(I, T, F, T)
    base = argmax(objs, fork(multiply, height, width))
    basecol = color(base)
    cols = remove(ZERO, palette(I))
    minor = other(cols, basecol)
    B = backdrop(base)
    dul = add(UP, LEFT)
    dur = add(UP, RIGHT)
    dll = add(DOWN, LEFT)
    drr = add(DOWN, RIGHT)
    ul = ulcorner(B)
    ur = urcorner(B)
    ll = llcorner(B)
    lr = lrcorner(B)
    A = asindices(I)
    rays = combine(
        combine(intersection(A, shoot(add(ul, dul), dul)),
                intersection(A, shoot(add(ur, dur), dur))),
        combine(intersection(A, shoot(add(ll, dll), dll)),
                intersection(A, shoot(add(lr, drr), drr))))
    O = underpaint(I, recolor(minor, rays))
    return O
