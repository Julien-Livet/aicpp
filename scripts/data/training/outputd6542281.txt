def dsl1(I):
    # Duplicate all non-background colored cells to the right, down, and down-right (preserving original colors)
    bg = mostcolor(I)
    P = palette(I)
    Pnb = remove(bg, P)
    nb = mapply(lbind(ofcolor, I), Pnb)
    objnb = toobject(nb, I)
    O = move(I, objnb, RIGHT)
    O = move(O, objnb, DOWN)
    O = move(O, objnb, UNITY)
    return O

def dsl2(I):
    # Expand around each 4: put 2 at up/left/up-left, 8 at down-right, and 3 at two steps down-right
    allidx = asindices(I)
    idx4 = ofcolor(I, FOUR)
    up = intersection(shift(idx4, UP), allidx)
    left = intersection(shift(idx4, LEFT), allidx)
    upleft = intersection(shift(idx4, NEG_UNITY), allidx)
    dr = intersection(shift(idx4, UNITY), allidx)
    ddr = intersection(shift(idx4, double(UNITY)), allidx)
    O = underfill(I, TWO, combine(combine(up, left), upleft))
    O = underfill(O, EIGHT, dr)
    O = underfill(O, THREE, ddr)
    return O

def dsl3(I):
    # Fill the bounding rectangle of all 3-cells with 3 (if any 3s exist)
    idx3 = ofcolor(I, THREE)
    rect = backdrop(idx3)
    O = underfill(I, THREE, rect)
    return O

def dsl4(I):
    # Union with its counter-diagonal mirror, but only for non-background colors
    bg = mostcolor(I)
    J = dmirror(I)
    P = palette(J)
    Pnb = remove(bg, P)
    nbJ = mapply(lbind(ofcolor, J), Pnb)
    objJ = toobject(nbJ, J)
    O = underpaint(I, objJ)
    return O

def dsl5(I):
    # Morphological expansion using global min/max non-bg colors around all non-bg cells
    bg = mostcolor(I)
    P = palette(I)
    Pnb = remove(bg, P)
    if size(Pnb) == 0:
        return I
    a = minimum(Pnb)
    d = maximum(Pnb)
    allidx = asindices(I)
    nb = mapply(lbind(ofcolor, I), Pnb)
    t_ul = combine(combine(shift(nb, NEG_UNITY), shift(nb, UP)), shift(nb, LEFT))
    t_ul = intersection(t_ul, allidx)
    t_r = intersection(shift(nb, RIGHT), allidx)
    t_d = intersection(shift(nb, DOWN), allidx)
    t_dr = intersection(shift(nb, UNITY), allidx)
    O = underfill(I, a, t_ul)
    O = underfill(O, a, t_r)
    O = underfill(O, a, t_d)
    O = underfill(O, d, t_dr)
    return O
