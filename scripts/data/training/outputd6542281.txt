def dsl1(I):
    bg = mostcolor(I)
    P = palette(I)
    Pnb = remove(bg, P)
    a = minimum(Pnb)
    d = maximum(Pnb)
    Ptmp = remove(d, Pnb)
    b = maximum(Ptmp)
    idx_b = ofcolor(I, b)
    allidx = asindices(I)
    ul = shift(idx_b, NEG_UNITY)
    up = shift(idx_b, UP)
    left = shift(idx_b, LEFT)
    patch1 = combine(combine(ul, up), left)
    patch1 = intersection(patch1, allidx)
    O = underfill(I, a, patch1)
    right = shift(idx_b, RIGHT)
    down = shift(idx_b, DOWN)
    patch2a = combine(right, down)
    patch2a = intersection(patch2a, allidx)
    O = underfill(O, a, patch2a)
    dr = shift(idx_b, UNITY)
    patchd = intersection(dr, allidx)
    O = underfill(O, d, patchd)
    return O

def dsl3(I):
    bg = mostcolor(I)
    P = palette(I)
    Pnb = remove(bg, P)
    a = minimum(Pnb)
    d = maximum(Pnb)
    b = maximum(remove(d, Pnb))
    idx_b = ofcolor(I, b)
    allidx = asindices(I)
    # complete T1 around b
    t1 = combine(combine(shift(idx_b, NEG_UNITY), shift(idx_b, UP)), shift(idx_b, LEFT))
    t1 = intersection(t1, allidx)
    O = underfill(I, a, t1)
    # place a to the right/down of b and d on the diagonal
    ra = intersection(shift(idx_b, RIGHT), allidx)
    da = intersection(shift(idx_b, DOWN), allidx)
    O = underfill(O, a, combine(ra, da))
    rd = intersection(shift(idx_b, UNITY), allidx)
    O = underfill(O, d, rd)
    return O
