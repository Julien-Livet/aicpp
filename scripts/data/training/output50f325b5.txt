def dsl1(I):
    idx3 = ofcolor(I, THREE)
    z = ofcolor(I, ZERO)
    adjz = combine(combine(shift(z, UP), shift(z, DOWN)), combine(shift(z, LEFT), shift(z, RIGHT)))
    to_change = intersection(idx3, adjz)
    O = fill(I, EIGHT, to_change)
    return O

def dsl2(I):
    idx3 = ofcolor(I, THREE)
    has_left3 = shift(idx3, RIGHT)
    has_right3 = shift(idx3, LEFT)
    has_up3 = shift(idx3, DOWN)
    has_down3 = shift(idx3, UP)
    horiz_end = intersection(idx3, combine(difference(has_left3, has_right3), difference(has_right3, has_left3)))
    vert_end = intersection(idx3, combine(difference(has_up3, has_down3), difference(has_down3, has_up3)))
    to_change = combine(horiz_end, vert_end)
    O = fill(I, EIGHT, to_change)
    return O
