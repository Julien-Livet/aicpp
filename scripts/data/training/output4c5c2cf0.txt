def dsl1(I):
    objs = objects(I, T, T, T)
    xo = first(totuple(sizefilter(objs, FIVE)))
    obj = other(objs, xo)
    dv = double(subtract(center(xo), center(obj)))
    off_h = multiply(dv, astuple(ONE, ZERO))
    off_v = multiply(dv, astuple(ZERO, ONE))
    vobj = shift(vmirror(obj), off_v)
    hobj = shift(hmirror(obj), off_h)
    both = shift(vmirror(hmirror(obj)), dv)
    O = underpaint(I, vobj)
    O = underpaint(O, hobj)
    O = underpaint(O, both)
    return O

def dsl2(I):
    objs = objects(I, T, T, T)
    t = totuple(objs)
    a = first(t)
    b = last(t)
    xa = combine(corners(a), initset(center(a)))
    isx = equality(toindices(a), xa)
    xo = branch(isx, a, b)
    obj = other(objs, xo)
    dv = double(subtract(center(xo), center(obj)))
    vobj = shift(vmirror(obj), multiply(dv, astuple(ZERO, ONE)))
    hobj = shift(hmirror(obj), multiply(dv, astuple(ONE, ZERO)))
    both = shift(vmirror(hmirror(obj)), dv)
    O = underpaint(I, vobj)
    O = underpaint(O, hobj)
    O = underpaint(O, both)
    return O

def dsl4(I):
    objs = objects(I, T, T, T)
    # robust pivot pick: prefer unique size-5 object, else fall back to corner+center pattern
    maybe = sizefilter(objs, FIVE)
    has5 = positive(size(maybe))
    xo = branch(has5, first(totuple(maybe)),
                branch(equality(toindices(first(totuple(objs))),
                                combine(corners(first(totuple(objs))), initset(center(first(totuple(objs)))))),
                       first(totuple(objs)),
                       last(totuple(objs)))))
    obj = other(objs, xo)
    dv = double(subtract(center(xo), center(obj)))
    vobj = shift(vmirror(obj), multiply(dv, astuple(ZERO, ONE)))
    hobj = shift(hmirror(obj), multiply(dv, astuple(ONE, ZERO)))
    both = shift(vmirror(hmirror(obj)), dv)
    O = paint(I, vobj)
    O = paint(O, hobj)
    O = paint(O, both)
    return O

def dsl5(I):
    objs = objects(I, T, T, T)
    # choose pivot by minimal bounding box (diamond is compact 3x3)
    xo = argmin(objs, compose(width, backdrop))
    obj = other(objs, xo)
    dv = double(subtract(center(xo), center(obj)))
    off_h = multiply(dv, astuple(ONE, ZERO))
    off_v = multiply(dv, astuple(ZERO, ONE))
    O = I
    O = underpaint(O, shift(hmirror(obj), off_h))
    O = underpaint(O, shift(vmirror(obj), off_v))
    O = underpaint(O, shift(vmirror(hmirror(obj)), dv))
    return O
