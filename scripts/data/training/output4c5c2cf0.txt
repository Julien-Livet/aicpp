def dsl1(I):
    objs = objects(I, T, T, T)
    tup = totuple(objs)
    a = first(tup)
    b = last(tup)
    xa = combine(corners(a), initset(center(a)))
    isx = equality(toindices(a), xa)
    xobj = branch(isx, a, b)
    obj = other(objs, xobj)
    cx = center(xobj)
    co = center(obj)
    pos = subtract(cx, co)
    dv = double(pos)
    off_v = multiply(dv, astuple(ZERO, ONE))
    off_h = multiply(dv, astuple(ONE, ZERO))
    vobj = shift(vmirror(obj), off_v)
    hobj = shift(hmirror(obj), off_h)
    dobj = shift(rot180(obj), dv)
    O = paint(I, vobj)
    O = paint(O, hobj)
    O = paint(O, dobj)
    return O
