def dsl1(I):
    objs7 = colorfilter(objects(I, T, F, T), SEVEN)
    centers = apply(center, objs7)
    eights = mapply(compose(lbind(recolor, EIGHT), initset), centers)
    upr = apply(urcorner, objs7)
    lwr = apply(lrcorner, objs7)
    edges = mpapply(connect, upr, lwr)
    v2 = recolor(TWO, merge(edges))
    G = paint(I, merge(eights))
    O = underpaint(G, v2)
    return O

def dsl2(I):
    objs7 = colorfilter(objects(I, T, F, T), SEVEN)
    centers = apply(center, objs7)
    # vertical columns at center columns
    cols = apply(vfrontier, centers)
    col2s = apply(lbind(recolor, TWO), cols)
    # horizontal 3-length 2s centered on center
    l1 = apply(lbind(add, LEFT), centers)
    r1 = apply(lbind(add, RIGHT), centers)
    hseg = mpapply(connect, l1, r1)
    h2s = mapply(lbind(recolor, TWO), hseg)
    # side markers 4 and 3 at +/-2
    l2 = apply(lbind(add, double(LEFT)), centers)
    r2 = apply(lbind(add, double(RIGHT)), centers)
    fours = mapply(compose(lbind(recolor, FOUR), initset), l2)
    threes = mapply(compose(lbind(recolor, THREE), initset), r2)
    # center highlights
    eights = mapply(compose(lbind(recolor, EIGHT), initset), centers)
    G = paint(I, merge(eights))
    H = underpaint(G, merge(col2s))
    J = underpaint(H, merge(h2s))
    K = underpaint(J, merge(fours))
    O = underpaint(K, merge(threes))
    return O
