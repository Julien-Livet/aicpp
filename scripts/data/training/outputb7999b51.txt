def dsl1(I):
    cols = remove(ZERO, palette(I))
    comp = compose(invert, lbind(colorcount, I))
    ordered = order(cols, comp)
    hfun = compose(height, lbind(ofcolor, I))
    nrows = valmax(ordered, hfun)
    ncols = size(ordered)
    base = canvas(ZERO, astuple(nrows, ncols))
    idxs = interval(ZERO, ncols, ONE)
    pairs = pair(idxs, ordered)
    get_j = first
    get_c = last
    startpt = compose(lbind(astuple, ZERO), get_j)
    endrow = compose(decrement, compose(hfun, get_c))
    endpt = fork(astuple, endrow, get_j)
    line = fork(connect, startpt, endpt)
    objfun = fork(recolor, get_c, line)
    objs = apply(objfun, pairs)
    merged = merge(objs)
    O = paint(base, merged)
    return O

def dsl2(I):
    cols = remove(ZERO, palette(I))
    comp = compose(invert, lbind(colorcount, I))
    ordered = order(cols, comp)
    hfun = compose(height, lbind(ofcolor, I))
    nrows = valmax(ordered, hfun)
    ncols = size(ordered)
    base = canvas(ZERO, astuple(nrows, ncols))
    idxs = interval(ZERO, ncols, ONE)
    pairs = pair(idxs, ordered)
    startpt = compose(lbind(astuple, ZERO), first)
    endrow = compose(decrement, compose(hfun, last))
    endpt = fork(astuple, endrow, first)
    seg = fork(connect, startpt, endpt)
    colobj = fork(recolor, last, seg)
    objs = apply(colobj, pairs)
    merged = merge(objs)
    O = underpaint(base, merged)
    return O

def dsl3(I):
    nz = remove(ZERO, palette(I))
    area_desc = compose(invert, lbind(colorcount, I))
    orderc = order(nz, area_desc)
    hgt = compose(height, lbind(ofcolor, I))
    rows = valmax(orderc, hgt)
    cols = size(orderc)
    out = canvas(ZERO, astuple(rows, cols))
    js = interval(ZERO, cols, ONE)
    pr = pair(js, orderc)
    st = compose(lbind(astuple, ZERO), first)
    er = compose(decrement, compose(hgt, last))
    en = fork(astuple, er, first)
    colline = fork(connect, st, en)
    colobj = fork(recolor, last, colline)
    pieces = apply(colobj, pr)
    union = merge(pieces)
    O = paint(out, union)
    return O
