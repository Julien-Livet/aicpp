def dsl1(I):
    twos = ofcolor(I, TWO)
    twoobj = toobject(twos, I)
    inside = inbox(twoobj)
    W = width(I)
    top = crop(I, astuple(ZERO, ZERO), astuple(ONE, W))
    topobjs = fgpartition(top)
    top2 = colorfilter(topobjs, TWO)
    topnz = difference(topobjs, top2)
    fdown = fork(recolor, color, compose(lbind(intersection, inside), compose(rbind(shoot, DOWN), ulcorner)))
    topset = mapply(fdown, topnz)
    topobj = merge(topset)
    H = height(I)
    startb = astuple(subtract(H, ONE), ZERO)
    bot = crop(I, startb, astuple(ONE, W))
    botobjs = fgpartition(bot)
    bot2 = colorfilter(botobjs, TWO)
    botnz = difference(botobjs, bot2)
    fup = fork(recolor, color, compose(lbind(intersection, inside), compose(lbind(shift, startb), compose(rbind(shoot, UP), ulcorner))))
    botset = mapply(fup, botnz)
    botobj = merge(botset)
    lines = combine(topobj, botobj)
    O = underpaint(I, lines)
    return O

def dsl3(I):
    twos = ofcolor(I, TWO)
    twoobj = toobject(twos, I)
    inside = inbox(twoobj)
    W = width(I)
    H = height(I)
    top = crop(I, astuple(ZERO, ZERO), astuple(ONE, W))

    fdown = fork(recolor, color, compose(lbind(intersection, inside), compose(rbind(shoot, DOWN), ulcorner)))
    topobj = merge(mapply(fdown, topnz))
    startb = astuple(subtract(H, ONE), ZERO)
    bot = crop(I, startb, astuple(ONE, W))
    botobjs = fgpartition(bot)
    botnz = difference(botobjs, colorfilter(botobjs, TWO))
    fup = fork(recolor, color, compose(lbind(intersection, inside), compose(lbind(shift, startb), compose(rbind(shoot, UP), ulcorner))))
    botobj = merge(mapply(fup, botnz))
    left = crop(I, astuple(ZERO, ZERO), astuple(H, ONE))
    lobjs = fgpartition(left)
    lnz = difference(lobjs, colorfilter(lobjs, TWO))
    fright = fork(recolor, color, compose(lbind(intersection, inside), compose(rbind(shoot, RIGHT), ulcorner)))
    lobj = merge(mapply(fright, lnz))
    startr = astuple(ZERO, subtract(W, ONE))
    right = crop(I, startr, astuple(H, ONE))
    robjs = fgpartition(right)
    rnz = difference(robjs, colorfilter(robjs, TWO))
    fleft = fork(recolor, color, compose(lbind(intersection, inside), compose(lbind(shift, startr), compose(rbind(shoot, LEFT), ulcorner))))
    robj = merge(mapply(fleft, rnz))
    lines = merge((combine(topobj, botobj), combine(lobj, robj)))
    O = underpaint(I, lines)
    return O
