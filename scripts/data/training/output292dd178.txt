def dsl1(I):
    objs = objects(I, T, F, T)
    onesobjs = colorfilter(objs, ONE)
    rf = compose(lbind(mapply, hfrontier), toindices)
    cf = compose(lbind(mapply, vfrontier), toindices)
    rcfunc = fork(lbind(prapply, intersection), rf, cf)
    rcs = mapply(rcfunc, onesobjs)
    O = underfill(I, TWO, rcs)
    return O

def dsl3(I):
    ones = ofcolor(I, ONE)
    # partition 1s into connected components
    objs = objects(fill(canvas(ZERO, shape(I)), ONE, ones), T, F, T)
    rf = compose(lbind(mapply, hfrontier), toindices)
    cf = compose(lbind(mapply, vfrontier), toindices)
    rcfunc = fork(lbind(prapply, intersection), rf, cf)
    rcs = mapply(rcfunc, objs)
    O = underfill(I, TWO, rcs)
    return O

def dsl4(I):
    objs = objects(I, T, F, T)
    onesobjs = colorfilter(objs, ONE)
    rows = mapply(compose(lbind(mapply, hfrontier), toindices), onesobjs)
    cols = mapply(compose(lbind(mapply, vfrontier), toindices), onesobjs)
    rcs = prapply(intersection, rows, cols)
    O = underfill(I, TWO, rcs)
    return O

def dsl5(I):
    # Same as dsl1 but compute background explicitly (robustness)
    bg = mostcolor(I)
    objs = objects(I, T, F, T)
    onesobjs = colorfilter(objs, ONE)
    rf = compose(lbind(mapply, hfrontier), toindices)
    cf = compose(lbind(mapply, vfrontier), toindices)
    rcfunc = fork(lbind(prapply, intersection), rf, cf)
    rcs = mapply(rcfunc, onesobjs)
    rc_bg = intersection(rcs, ofcolor(I, bg))
    O = fill(I, TWO, rc_bg)
    return O
