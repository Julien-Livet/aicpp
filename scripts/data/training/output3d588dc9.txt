def dsl1(I):
    objs = objects(I, T, F, T)
    zeros = colorfilter(objs, ZERO)
    sel = argmax(zeros, height)
    zinds = toindices(sel)
    coreL = intersection(zinds, shift(zinds, LEFT))
    coreR = intersection(zinds, shift(zinds, RIGHT))
    core = combine(coreL, coreR)
    J = cover(I, sel)
    J = paint(J, recolor(ZERO, core))
    mid = halve(width(I))
    lbord = intersection(core, vfrontier(ulcorner(sel)))
    rbord = intersection(core, vfrontier(urcorner(sel)))
    side = branch(greater(mid, rightmost(sel)), rbord, lbord)
    O = fill(J, SIX, side)
    return O
