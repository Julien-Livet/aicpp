def dsl1(I):
    bg = mostcolor(I)
    objs = objects(I, T, F, T)
    squares = sizefilter(objs, 9)
    seq = order(squares, leftmost)
    L = first(seq)
    R = last(seq)
    M = first(remove(R, remove(L, seq)))
    A = subgrid(L, I)
    B = subgrid(M, I)
    C = subgrid(R, I)
    sepcol = canvas(bg, astuple(height(A), 1))
    row1 = hconcat(hconcat(hconcat(B, sepcol), A), sepcol)
    row1 = hconcat(row1, C)
    seprow = canvas(bg, astuple(1, width(row1)))
    row2 = hconcat(hconcat(hconcat(C, sepcol), B), sepcol)
    row2 = hconcat(row2, A)
    row3 = hconcat(hconcat(hconcat(A, sepcol), C), sepcol)
    row3 = hconcat(row3, B)
    O = vconcat(row1, seprow)
    O = vconcat(O, row2)
    O = vconcat(O, seprow)
    O = vconcat(O, row3)
    O = vconcat(O, seprow)
    O = vconcat(O, seprow)
    return O
