def dsl1(I):
    # Stack the two connected components of each color bottom-aligned with a 1-row gap, in two vertical bands (left and right halves)
    hI = height(I); wI = width(I); mid = divide(wI, TWO)
    cols = remove(ZERO, palette(I)); c1 = first(cols); c2 = other(cols, c1)
    p1 = ofcolor(I, c1); p2 = ofcolor(I, c2)
    l1 = leftmost(p1); l2 = leftmost(p2)
    # assign left/right by which color is more to the left
    cL = branch(greater(l1, l2), c2, c1); cR = branch(greater(l1, l2), c1, c2)
    allobjs = objects(I, T, F, T)
    oLset = colorfilter(allobjs, cL); oRset = colorfilter(allobjs, cR)
    topL = argmin(oLset, uppermost); botL = argmax(oLset, lowermost)
    topR = argmin(oRset, uppermost); botR = argmax(oRset, lowermost)
    hTopL = height(subgrid(topL, I)); hBotL = height(subgrid(botL, I))
    hTopR = height(subgrid(topR, I)); hBotR = height(subgrid(botR, I))
    off_bot_L = subtract(hI, hBotL)
    off_top_L = subtract(off_bot_L, add(hTopL, ONE))
    off_bot_R = subtract(hI, hBotR)
    off_top_R = subtract(off_bot_R, add(hTopR, ONE))
    base = canvas(ZERO, shape(I))
    tmp = paint(base, shift(normalize(topL), astuple(off_top_L, ZERO)))
    tmp = paint(tmp, shift(normalize(botL), astuple(off_bot_L, ZERO)))
    tmp = paint(tmp, shift(normalize(topR), astuple(off_top_R, mid)))
    O = paint(tmp, shift(normalize(botR), astuple(off_bot_R, mid)))
    return O

def dsl3(I):
    # Treat each color as a whole; bottom-align each color𠏋 full shape into left and right halves (separator at mid column)
    hI = height(I); wI = width(I); mid = divide(wI, TWO)
    cols = remove(ZERO, palette(I)); c1 = first(cols); c2 = other(cols, c1)
    p1 = ofcolor(I, c1); p2 = ofcolor(I, c2)
    l1 = leftmost(p1); l2 = leftmost(p2)
    cL = branch(greater(l1, l2), c2, c1); cR = branch(greater(l1, l2), c1, c2)
    oL = toobject(ofcolor(I, cL), I); oR = toobject(ofcolor(I, cR), I)
    hL = height(subgrid(oL, I)); hR = height(subgrid(oR, I))
    offL = subtract(hI, hL); offR = subtract(hI, hR)
    base = canvas(ZERO, shape(I))
    tmp = paint(base, shift(normalize(oL), astuple(offL, ZERO)))
    O = paint(tmp, shift(normalize(oR), astuple(offR, mid)))
    return O

def dsl5(I):
    # Build left/right halves independently by stacking each color𠏋 two components bottom-aligned, then concatenate halves
    hI = height(I); wI = width(I); mid = divide(wI, TWO)
    cols = remove(ZERO, palette(I)); c1 = first(cols); c2 = other(cols, c1)
    p1 = ofcolor(I, c1); p2 = ofcolor(I, c2)
    l1 = leftmost(p1); l2 = leftmost(p2)
    cL = branch(greater(l1, l2), c2, c1); cR = branch(greater(l1, l2), c1, c2)
    allobjs = objects(I, T, F, T)
    oLset = colorfilter(allobjs, cL); oRset = colorfilter(allobjs, cR)
    topL = argmin(oLset, uppermost); botL = argmax(oLset, lowermost)
    topR = argmin(oRset, uppermost); botR = argmax(oRset, lowermost)
    hTopL = height(subgrid(topL, I)); hBotL = height(subgrid(botL, I))
    hTopR = height(subgrid(topR, I)); hBotR = height(subgrid(botR, I))
    off_bot_L = subtract(hI, hBotL); off_top_L = subtract(off_bot_L, add(hTopL, ONE))
    off_bot_R = subtract(hI, hBotR); off_top_R = subtract(off_bot_R, add(hTopR, ONE))
    left_canvas = canvas(ZERO, astuple(hI, mid))
    right_canvas = canvas(ZERO, astuple(hI, mid))
    left_side = paint(left_canvas, shift(normalize(topL), astuple(off_top_L, ZERO)))
    left_side = paint(left_side, shift(normalize(botL), astuple(off_bot_L, ZERO)))
    right_side = paint(right_canvas, shift(normalize(topR), astuple(off_top_R, ZERO)))
    right_side = paint(right_side, shift(normalize(botR), astuple(off_bot_R, ZERO)))
    O = hconcat(left_side, right_side)
    return O
