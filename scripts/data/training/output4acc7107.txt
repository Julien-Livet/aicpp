def dsl1(I):
    cset = remove(ZERO, palette(I))
    c1 = first(cset)
    c2 = other(cset, c1)
    p1 = ofcolor(I, c1)
    p2 = ofcolor(I, c2)
    o1 = recolor(c1, p1)
    o2 = recolor(c2, p2)
    sg1 = subgrid(o1, I)
    sg2 = subgrid(o2, I)
    hI = height(I)
    w1 = width(sg1)
    h1 = height(sg1)
    h2 = height(sg2)
    off1 = astuple(subtract(hI, h1), ZERO)
    start2 = add(w1, ONE)
    off2 = astuple(subtract(hI, h2), start2)
    n1 = normalize(o1)
    n2 = normalize(o2)
    base = canvas(ZERO, shape(I))
    tmp = paint(base, shift(n1, off1))
    O = paint(tmp, shift(n2, off2))
    return O

def dsl5(I):
    x = compress(I)
    obj = asobject(x)
    H = height(I)
    h = height(x)
