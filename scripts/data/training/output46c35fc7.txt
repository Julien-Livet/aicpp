def dsl1(I):
    O = I
    for obj in totuple(objects(I, F, T, T)):
        sub = subgrid(obj, I)
        idxs = asindices(sub)
        ring = box(idxs)
        crn = corners(idxs)
        edg = difference(ring, crn)
        cw = rot90(sub)
        ccw = rot270(sub)
        oc = toobject(crn, ccw)  # corners from CCW
        oe = toobject(edg, cw)   # edges from CW
        base = canvas(SEVEN, shape(sub))
        base = paint(base, oc)
        base = paint(base, oe)
        O = cover(O, obj)
        O = move(O, asobject(base), ulcorner(obj))
    return O
