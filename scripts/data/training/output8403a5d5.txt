def dsl1(I):
    objs = fgpartition(I)
    seed = first(objs)
    k = color(seed)
    j0 = leftmost(seed)
    H = height(I)
    lastrow = subtract(H, ONE)
    base = connect(astuple(ZERO, j0), astuple(lastrow, j0))
    c1 = shift(base, ZERO_BY_TWO)
    c2 = shift(c1, ZERO_BY_TWO)
    c3 = shift(c2, ZERO_BY_TWO)
    c4 = shift(c3, ZERO_BY_TWO)
    cols = combine(combine(base, c1), combine(c2, combine(c3, c4)))
    cols = intersection(cols, asindices(I))
    stripes = recolor(k, cols)
    j1 = add(j0, ONE)
    j3 = add(j0, THREE)
    j5 = add(j0, FIVE)
    j7 = add(j0, SEVEN)
    j9 = add(j0, NINE)
    p1 = initset(astuple(ZERO, j1))
    p2 = initset(astuple(lastrow, j3))
    p3 = initset(astuple(ZERO, j5))
    p4 = initset(astuple(lastrow, j7))
    p5 = initset(astuple(ZERO, j9))
    fivepatch = combine(combine(p1, p2), combine(p3, combine(p4, p5)))
    fivepatch = intersection(fivepatch, asindices(I))
    fives = recolor(FIVE, fivepatch)
    O = paint(canvas(ZERO, shape(I)), stripes)
    O = paint(O, fives)
    return O

def dsl3(I):
    objs = fgpartition(I)
    seed = first(objs)
    k = color(seed)
    j0 = leftmost(seed)
    H = height(I)
    lastrow = subtract(H, ONE)
    col0 = connect(astuple(ZERO, j0), astuple(lastrow, j0))
    cols = intersection(combine(col0, shift(col0, ZERO_BY_TWO)), asindices(I))
    cols = intersection(combine(cols, shift(cols, ZERO_BY_TWO)), asindices(I))
    stripes = recolor(k, cols)
    j1 = add(j0, ONE)
    j3 = add(j0, THREE)
    j5 = add(j0, FIVE)
    j7 = add(j0, SEVEN)
    j9 = add(j0, NINE)
    top = ZERO
    bot = lastrow
    fivepatch = combine(initset(astuple(top, j1)), initset(astuple(bot, j3)))
    fivepatch = combine(fivepatch, initset(astuple(top, j5)))
    fivepatch = combine(fivepatch, initset(astuple(bot, j7)))
    fivepatch = combine(fivepatch, initset(astuple(top, j9)))
    fivepatch = intersection(fivepatch, asindices(I))
    fives = recolor(FIVE, fivepatch)
    O = paint(canvas(ZERO, shape(I)), stripes)
    O = paint(O, fives)
    return O
