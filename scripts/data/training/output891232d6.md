def dsl1(I):
    sev = colorfilter(objects(I, T, F, T), SEVEN)
    cents = apply(center, sev)
    obj8 = recolor(EIGHT, merge(apply(initset, cents)))
    J = paint(I, obj8)
    starts = apply(rbind(add, DOWN), cents)
    vpaths = merge(apply(rbind(shoot, DOWN), starts))
    v2 = recolor(TWO, vpaths)
    O = underpaint(J, v2)
    return O

def dsl2(I):
    sev = colorfilter(objects(I, T, F, T), SEVEN)
    cents = apply(center, sev)
    J = paint(I, recolor(EIGHT, merge(apply(initset, cents))))
    vecL = multiply(RIGHT, NEG_TWO)
    vecR = multiply(RIGHT, TWO)
    left = apply(rbind(add, vecL), cents)
    right = apply(rbind(add, vecR), cents)
    belowLeft = apply(rbind(add, DOWN), left)
    belowRight = apply(rbind(add, DOWN), right)
    hseg = merge(papply(connect, belowLeft, belowRight))
    line2 = recolor(TWO, hseg)
    Lend4 = recolor(FOUR, merge(apply(initset, belowLeft)))
    Rend3 = recolor(THREE, merge(apply(initset, belowRight)))
    parts = merge(combine(initset(line2), combine(initset(Lend4), initset(Rend3))))
    O = underpaint(J, parts)
    return O

def dsl3(I):

def dsl4(I):
    sev = colorfilter(objects(I, T, F, T), SEVEN)
    boxes = apply(box, sev)
    uls = apply(ulcorner, boxes)
    urs = apply(urcorner, boxes)
    cents = apply(center, sev)
    obj4 = recolor(FOUR, merge(apply(initset, uls)))
    obj3 = recolor(THREE, merge(apply(initset, urs)))
    obj8 = recolor(EIGHT, merge(apply(initset, cents)))
    parts = merge(combine(initset(obj4), combine(initset(obj3), initset(obj8))))
    O = underpaint(I, parts)
    return O
