def dsl1(I):
    # Full gravitate of interior left/right colors to their respective border columns
    w = width(I)
    leftc = index(I, astuple(ZERO, ZERO))
    rightc = index(I, astuple(ZERO, subtract(w, ONE)))
    leftcol = shoot(astuple(ZERO, ZERO), DOWN)
    rightcol = shoot(astuple(ZERO, subtract(w, ONE)), DOWN)
    # move all interior left-color cells fully toward the left column
    allL = ofcolor(I, leftc)
    internalL = difference(allL, leftcol)
    objL = toobject(internalL, I)
    dirL = gravitate(internalL, leftcol)
    g1 = move(I, objL, dirL)
    # then move all interior right-color cells fully toward the right column
    allR = ofcolor(g1, rightc)
    internalR = difference(allR, rightcol)
    objR = toobject(internalR, g1)
    dirR = gravitate(internalR, rightcol)
    O = move(g1, objR, dirR)
    return O

def dsl2(I):
    # Fill interior (excluding first and last columns) with the most common color, keep border columns intact
    dims = shape(I)
    m = mostcolor(I)
    w = width(I)
    leftcol = shoot(astuple(ZERO, ZERO), DOWN)
    rightcol = shoot(astuple(ZERO, subtract(w, ONE)), DOWN)
    allinds = asindices(I)
    interior = difference(allinds, combine(leftcol, rightcol))
    O = canvas(m, dims)
    O = paint(O, toobject(leftcol, I))
    O = paint(O, toobject(rightcol, I))
    # Ensure interior is set to m explicitly (in case paint affected it)
    O = fill(O, m, interior)
    return O

def dsl3(I):
    # Suppress interior right-color first, pack left-color fully to the left border, then pack right-color to the right border
    w = width(I)
    leftc = index(I, astuple(ZERO, ZERO))
    rightc = index(I, astuple(ZERO, subtract(w, ONE)))
    leftcol = shoot(astuple(ZERO, ZERO), DOWN)
    rightcol = shoot(astuple(ZERO, subtract(w, ONE)), DOWN)
    m = mostcolor(I)
    # remove interior right-color to reduce conflicts
    allR0 = ofcolor(I, rightc)
    internalR0 = difference(allR0, rightcol)
    g0 = fill(I, m, internalR0)
    # gravitate interior left-color fully to left column
    allL = ofcolor(g0, leftc)
    internalL = difference(allL, leftcol)
    objL = toobject(internalL, g0)
    dirL = gravitate(internalL, leftcol)
    g1 = move(g0, objL, dirL)
    # gravitate interior right-color (recomputed) fully to right column
    allR = ofcolor(g1, rightc)
    internalR = difference(allR, rightcol)
    objR = toobject(internalR, g1)
    dirR = gravitate(internalR, rightcol)
    O = move(g1, objR, dirR)
    return O

def dsl4(I):
    # Vertical packing: move interior left-border color up to top row, right-border color down to bottom row
    h = height(I)
    w = width(I)
    leftc = index(I, astuple(ZERO, ZERO))
    rightc = index(I, astuple(ZERO, subtract(w, ONE)))
    toprow = shoot(astuple(ZERO, ZERO), RIGHT)
    bottomrow = shoot(astuple(subtract(h, ONE), ZERO), RIGHT)
    # move all non-top-row left-color cells up
    allL = ofcolor(I, leftc)
    internalLt = difference(allL, toprow)
    objLt = toobject(internalLt, I)
    dirUp = gravitate(internalLt, toprow)
    g1 = move(I, objLt, dirUp)
    # move all non-bottom-row right-color cells down
    allR = ofcolor(g1, rightc)
    internalRb = difference(allR, bottomrow)
    objRb = toobject(internalRb, g1)
    dirDn = gravitate(internalRb, bottomrow)
    O = move(g1, objRb, dirDn)
    return O

def dsl5(I):
    # Interior reset to mostcolor, then reinsert/move interior left/right colors fully to their borders and paint over
    w = width(I)
    dims = shape(I)
    m = mostcolor(I)
    leftc = index(I, astuple(ZERO, ZERO))
    rightc = index(I, astuple(ZERO, subtract(w, ONE)))
    leftcol = shoot(astuple(ZERO, ZERO), DOWN)
    rightcol = shoot(astuple(ZERO, subtract(w, ONE)), DOWN)
    allinds = asindices(I)
    interior = difference(allinds, combine(leftcol, rightcol))
    # base canvas
    O = canvas(m, dims)
    O = paint(O, toobject(leftcol, I))
    O = paint(O, toobject(rightcol, I))
    # prepare interior left/right objects and move them fully toward their borders
    allL = ofcolor(I, leftc)
    internalL = difference(allL, leftcol)
    objL = toobject(internalL, I)
    dirL = gravitate(internalL, leftcol)
    movedL = move(canvas(ZERO, dims), objL, dirL)  # move on a blank canvas
    O = underpaint(O, asobject(movedL))  # paint moved left-color onto O
    allR = ofcolor(I, rightc)
    internalR = difference(allR, rightcol)
    objR = toobject(internalR, I)
    dirR = gravitate(internalR, rightcol)
    movedR = move(canvas(ZERO, dims), objR, dirR)
    O = underpaint(O, asobject(movedR))  # paint moved right-color onto O
    return O
