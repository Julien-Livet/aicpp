def dsl1(I):
    L = lefthalf(I)
    R = righthalf(I)
    x1 = replace(R, FIVE, FOUR)
    x2 = replace(x1, SEVEN, FIVE)
    O = hconcat(L, x2)
    return O

    xT = replace(T, FIVE, FOUR)

def dsl4(I):
    T = tophalf(I)
    B = bottomhalf(I)
    TL = lefthalf(T)
    TR = righthalf(T)
    BL = lefthalf(B)
    BR = righthalf(B)
    mTR = replace(replace(TR, FIVE, FOUR), SEVEN, FIVE)
    mBR = replace(replace(BR, FIVE, FOUR), SEVEN, FIVE)
    top = hconcat(TL, mTR)
    bottom = hconcat(BL, mBR)
    O = vconcat(top, bottom)
    return O
