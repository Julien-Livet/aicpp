def dsl1(I):
    # Overlay 4x4 blocks with precedence: C > D > A > B (fill holes with next)
    dims = astuple(FOUR, FOUR)
    a = crop(I, astuple(ZERO, ZERO), dims)
    b = crop(I, astuple(ZERO, FIVE), dims)
    c = crop(I, astuple(ZERO, add(TEN, ZERO)), dims)
    d = crop(I, astuple(ZERO, add(TEN, FIVE)), dims)
    z = canvas(ZERO, dims)
    y = underpaint(z, asobject(c))
    y = underpaint(y, asobject(d))
    y = underpaint(y, asobject(a))
    O = underpaint(y, asobject(b))
    return O

def dsl4(I):

    dims = astuple(FOUR, FOUR)
    a = crop(I, astuple(ZERO, ZERO), dims)
