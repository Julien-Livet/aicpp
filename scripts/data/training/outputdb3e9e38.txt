def dsl1(I):
    P = ofcolor(I, SEVEN)
    H = height(P)
    Ds = interval(ZERO, H, ONE)
    dirs = combine(initset(RIGHT), initset(LEFT))
    def fun(d, s):
        v1 = toivec(invert(d))
        v2 = multiply(s, d)
        dir = add(v1, v2)
        Sft = shift(P, dir)
        c = branch(even(d), SEVEN, EIGHT)
        return recolor(c, Sft)
    objs = prapply(fun, Ds, dirs)
    obj = merge(objs)
    O = paint(canvas(ZERO, shape(I)), obj)
    return O

def dsl2(I):
    P = ofcolor(I, SEVEN)
    H = height(P)
    Ds = interval(ZERO, H, ONE)
    Sset = combine(initset(RIGHT), initset(LEFT))
    def shiftfun(d, s):
        return shift(P, add(toivec(invert(d)), multiply(s, d)))
    evenDs = sfilter(Ds, even)
    oddDs = sfilter(Ds, compose(flip, even))
    evset = prapply(shiftfun, evenDs, Sset)
    odset = prapply(shiftfun, oddDs, Sset)
    ev = recolor(SEVEN, merge(evset))
    od = recolor(EIGHT, merge(odset))
    obj = combine(ev, od)
    O = paint(canvas(ZERO, shape(I)), obj)
    return O

def dsl4(I):

    def shifted(d, s):
        return shift(P, add(toivec(invert(d)), multiply(s, d)))
    parts = prapply(shifted, Ds, LR)
