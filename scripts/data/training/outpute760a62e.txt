def dsl1(I):
    p2 = ofcolor(I, TWO)
    p3 = ofcolor(I, THREE)
    e = ofcolor(I, EIGHT)
    vrects = mapply(compose(backdrop, vfrontier), p2)
    hrects = mapply(compose(backdrop, hfrontier), p3)
    vmask = difference(merge(vrects), e)
    hmask = difference(merge(hrects), e)
    s = intersection(vmask, hmask)
    o2 = recolor(TWO, difference(vmask, s))
    o3 = recolor(THREE, difference(hmask, s))
    o6 = recolor(SIX, s)
    x1 = underpaint(I, o2)
    x2 = underpaint(x1, o3)
    O = paint(x2, o6)
    return O

def dsl4(I):
    twos = ofcolor(I, TWO)
    threes = ofcolor(I, THREE)
    e = ofcolor(I, EIGHT)
    vbands = merge(mapply(compose(backdrop, vfrontier), twos))
    hbands = merge(mapply(compose(backdrop, hfrontier), threes))
    vmask = difference(vbands, e)
    hmask = difference(hbands, e)
    cross = intersection(vmask, hmask)
    o6 = recolor(SIX, cross)
    x = underpaint(I, recolor(TWO, vmask))
    x = underpaint(x, recolor(THREE, hmask))
    O = paint(x, o6)
    return O

def dsl5(I):
    e = ofcolor(I, EIGHT)
    cols = merge(mapply(compose(backdrop, vfrontier), ofcolor(I, TWO)))
    rows = merge(mapply(compose(backdrop, hfrontier), ofcolor(I, THREE)))
    cols = difference(cols, e)
    rows = difference(rows, e)
    both = intersection(cols, rows)
    o2 = recolor(TWO, difference(cols, both))
    o3 = recolor(THREE, difference(rows, both))
    o6 = recolor(SIX, both)
    O = paint(underpaint(underpaint(I, o2), o3), o6)
    return O
