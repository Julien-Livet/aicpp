def dsl1(I):
    cols = remove(ZERO, palette(I))
    f = lbind(ofcolor, I)
    isv = fork(greater, compose(height, f), compose(width, f))
    vset = mfilter(cols, isv)
    vcol = first(vset)
    hcol = other(cols, vcol)
    vp = ofcolor(I, vcol)
    hp = ofcolor(I, hcol)
    vb = backdrop(vp)
    hb = backdrop(hp)
    intr = intersection(vb, hb)
    rv = intersection(intr, vp)
    rh = intersection(intr, hp)
    x1 = fill(I, hcol, rv)
    O = fill(x1, vcol, rh)
    return O

def dsl2(I):
    cols = remove(ZERO, palette(I))
    f = lbind(ofcolor, I)
    hfun = compose(height, f)
    wfun = compose(width, f)
    comp = fork(subtract, hfun, wfun)
    vcol = argmax(cols, comp)
    hcol = other(cols, vcol)
    vp = ofcolor(I, vcol)
    hp = ofcolor(I, hcol)
    vb = backdrop(vp)
    hb = backdrop(hp)
    intr = intersection(vb, hb)
    rv = intersection(intr, vp)
    rh = intersection(intr, hp)
    x1 = fill(I, hcol, rv)
    O = fill(x1, vcol, rh)
    return O

def dsl3(I):
    cols = remove(ZERO, palette(I))
    f = lbind(ofcolor, I)
    isv = compose(portrait, f)
    vset = mfilter(cols, isv)
    vcol = first(vset)
    hcol = other(cols, vcol)
    vp = ofcolor(I, vcol)
    hp = ofcolor(I, hcol)
    intr = intersection(backdrop(vp), backdrop(hp))
    rv = intersection(intr, vp)
    rh = intersection(intr, hp)
    rv_obj = recolor(hcol, rv)
    rh_obj = recolor(vcol, rh)
    x1 = paint(I, rv_obj)
    O = paint(x1, rh_obj)
    return O
