    H = merge(apply(hfrontier, idx8))
    V = merge(apply(vfrontier, idx8))
    P = combine(H, V)
    zeros = ofcolor(I, ZERO)
    twos = ofcolor(I, TWO)
    Pz = intersection(P, zeros)
    Pt = intersection(P, twos)
    objZ = recolor(FOUR, toobject(Pz, I))
