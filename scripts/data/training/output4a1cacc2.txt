def dsl1(I):
    objs = fgpartition(I)
    obj = first(objs)
    i = uppermost(obj)
    j = leftmost(obj)
    H = height(I)
    W = width(I)
    topd = i
    botd = subtract(H, increment(i))
    leftd = j
    rightd = subtract(W, increment(j))
    Hm1 = subtract(H, ONE)
    Wm1 = subtract(W, ONE)
    rstar = branch(greater(topd, botd), Hm1, ZERO)
    cstar = branch(greater(leftd, rightd), Wm1, ZERO)
    p = astuple(i, j)
    q = astuple(rstar, cstar)
    s0 = initset(p)
    s = insert(q, s0)
    R = backdrop(s)
    col = color(obj)
    O = fill(I, col, R)
    return O
